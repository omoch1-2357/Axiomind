1//! # axm-ai: AI Opponent System for Poker
2//!
3//! Provides AI opponent implementations for Texas Hold'em poker gameplay.
4//! Supports multiple AI strategies with a common interface for decision-making.
5//!
6//! ## Core Components
7//!
8//! - [`AIOpponent`] - Trait defining the interface for AI decision-making
9//! - [`baseline`] - Baseline AI implementation for testing and comparison
10//! - [`create_ai`] - Factory function for creating AI opponents
11//!
12//! ## Quick Start
13//!
14//! ```rust
15//! use axm_ai::{create_ai, AIOpponent};
16//! use axm_engine::engine::Engine;
17//!
18//! // Create a baseline AI opponent
19//! let ai = create_ai("baseline");
20//!
21//! // Use the AI to make decisions during gameplay
22//! let mut engine = Engine::new(Some(42), 1);
23//! engine.shuffle();
24//! engine.deal_hand().expect("Failed to deal hand");
25//!
26//! let player_id = engine.current_player().expect("No current player");
27//! let action = ai.get_action(&engine, player_id);
28//! println!("AI chose action: {:?}", action);
29//! ```
30//!
31//! ## AI Types
32//!
33//! Currently supported AI types:
34//! - `"baseline"` - Simple baseline AI for testing and benchmarking
35

41/// Trait defining the interface for AI opponents in poker games.
42/// Implementors must provide methods for decision-making and identification.
43///
44/// # Required Methods
45///
46/// - [`get_action`](AIOpponent::get_action) - Determine the next action based on game state
47/// - [`name`](AIOpponent::name) - Return the AI's identifier/name
48///
49/// # Example Implementation
50///
51/// ```rust
52/// use axm_ai::AIOpponent;
53/// use axm_engine::engine::Engine;
54/// use axm_engine::player::PlayerAction;
55///
56/// struct MyAI;
57///
58/// impl AIOpponent for MyAI {
59///     fn get_action(&self, engine: &Engine, player_id: usize) -> PlayerAction {
60///         // Simple strategy: always check or call
61///         PlayerAction::Call
62///     }
63///
64///     fn name(&self) -> &str {
65///         "MyAI"
66///     }
67/// }
68/// ```
69pub trait AIOpponent: Send + Sync {
70    /// Determine the next action for the AI player based on current game state.
71    ///
72    /// # Arguments
73    ///
74    /// * `engine` - Reference to the game engine containing current state
75    /// * `player_id` - The player ID for which to determine an action (0 or 1)
76    ///
77    /// # Returns
78    ///
79    /// A `PlayerAction` representing the AI's chosen move
80    ///
81    /// # Example
82    ///
83    /// ```ignore
84    /// let action = ai.get_action(&engine, 0);
85    /// match action {
86    ///     PlayerAction::Call => println!("AI calls"),
87    ///     PlayerAction::Raise(amount) => println!("AI raises {}", amount),
88    ///     _ => println!("AI takes other action"),
89    /// }
90    /// ```
91    fn get_action(&self, engine: &Engine, player_id: usize) -> PlayerAction;
92
93    /// Return the name/identifier of this AI implementation.
94    ///
95    /// # Returns
96    ///
97    /// A string slice containing the AI's name
98    ///
99    /// # Example
100    ///
101    /// ```ignore
102    /// println!("Playing against: {}", ai.name());
103    /// ```
104    fn name(&self) -> &str;
105}
106
107/// Factory function to create AI opponents by type string.
108///
109/// # Arguments
110///
111/// * `ai_type` - String identifier for the AI type (e.g., "baseline")
112///
113/// # Returns
114///
115/// A boxed trait object implementing `AIOpponent`
116///
117/// # Supported AI Types
118///
119/// - `"baseline"` - Simple baseline AI for testing
120///
121/// # Example
122///
123/// ```rust
124/// use axm_ai::create_ai;
125///
126/// let ai = create_ai("baseline");
127/// assert_eq!(ai.name(), "BaselineAI");
128/// ```
129///
130/// # Panics
131///
132/// Panics if an unknown AI type is requested. Currently only "baseline" is supported.
133pub fn create_ai(ai_type: &str) -> Box<dyn AIOpponent> {
134    match ai_type {
135        "baseline" => Box::new(baseline::BaselineAI::new()),
136        _ => panic!("Unknown AI type: {}", ai_type),
137    }
