1//! AI policy evaluation command.
2//!
3//! This module provides functionality to evaluate AI policies head-to-head across multiple hands,
4//! comparing their performance metrics including win rates, chip deltas, and action distributions.

9use axiomind_engine::engine::Engine;
10use axiomind_engine::hand::{compare_hands, evaluate_hand};
11use axiomind_engine::logger::ActionRecord;

14/// Statistics tracked for AI evaluation comparison
15#[derive(Debug, Clone)]
16struct EvalStats {
17    hands_played: u32,
18    wins: u32,
19    losses: u32,
20    ties: u32,
21    total_chips_won: i64,
22    total_pot_size: u64,
23    folds: u32,
24    checks: u32,
25    calls: u32,
26    bets: u32,
27    raises: u32,
28    all_ins: u32,
29}
30
31impl EvalStats {
32    fn new() -> Self {
33        Self {
34            hands_played: 0,
35            wins: 0,
36            losses: 0,
37            ties: 0,
38            total_chips_won: 0,
39            total_pot_size: 0,
40            folds: 0,
41            checks: 0,
42            calls: 0,
43            bets: 0,
44            raises: 0,
45            all_ins: 0,
46        }
47    }
48
49    fn update_from_actions(&mut self, actions: &[ActionRecord], player_id: usize) {
50        for action in actions {
51            if action.player_id == player_id {
52                use axiomind_engine::player::PlayerAction;
53                match action.action {
54                    PlayerAction::Fold => self.folds += 1,
55                    PlayerAction::Check => self.checks += 1,
56                    PlayerAction::Call => self.calls += 1,
57                    PlayerAction::Bet(_) => self.bets += 1,
58                    PlayerAction::Raise(_) => self.raises += 1,
59                    PlayerAction::AllIn => self.all_ins += 1,
60                }
61            }
62        }
63    }
64
65    fn update_result(&mut self, won: bool, tied: bool, chip_delta: i64, pot: u32) {
66        self.hands_played += 1;
67        if tied {
68            self.ties += 1;
69        } else if won {
70            self.wins += 1;
71        } else {
72            self.losses += 1;
73        }
74        self.total_chips_won += chip_delta;
75        self.total_pot_size += pot as u64;
76    }
77
78    fn win_rate(&self) -> f64 {
79        if self.hands_played == 0 {
80            0.0
81        } else {
82            (self.wins as f64 / self.hands_played as f64) * 100.0
83        }
84    }
85
86    fn avg_chip_delta(&self) -> f64 {
87        if self.hands_played == 0 {
88            0.0
89        } else {
90            self.total_chips_won as f64 / self.hands_played as f64
91        }
92    }
93
94    fn avg_pot_size(&self) -> f64 {
95        if self.hands_played == 0 {
96            0.0
97        } else {
98            self.total_pot_size as f64 / self.hands_played as f64
99        }
100    }
101
102    fn action_percentage(&self, count: u32) -> f64 {
103        let total_actions =
104            self.folds + self.checks + self.calls + self.bets + self.raises + self.all_ins;
105        if total_actions == 0 {
106            0.0
107        } else {
108            (count as f64 / total_actions as f64) * 100.0
109        }
110    }
111}

113/// Evaluates two AI policies head-to-head across multiple hands.
114///
115/// # Arguments
116///
117/// * `ai_a` - First AI policy identifier
118/// * `ai_b` - Second AI policy identifier
119/// * `hands` - Number of hands to play
120/// * `seed` - Optional seed for reproducibility
121/// * `out` - Output stream for evaluation results
122///
123/// # Returns
124///
125/// `Result<(), CliError>`: `Ok(())` when evaluation completes successfully.
126pub fn handle_eval_command(
127    ai_a: &str,
128    ai_b: &str,
129    hands: u32,
130    seed: Option<u64>,
131    out: &mut dyn Write,
132) -> Result<(), CliError> {
133    // Create AI instances
134    let ai_policy_a = match std::panic::catch_unwind(|| create_ai(ai_a)) {
135        Ok(ai) => ai,
136        Err(_) => {
137            return Err(CliError::InvalidInput(format!("Unknown AI type: {}", ai_a)));
138        }
139    };
140
141    let ai_policy_b = match std::panic::catch_unwind(|| create_ai(ai_b)) {
142        Ok(ai) => ai,
143        Err(_) => {
144            return Err(CliError::InvalidInput(format!("Unknown AI type: {}", ai_b)));
145        }
146    };
147
148    // Initialize statistics
149    let mut stats_a = EvalStats::new();
150    let mut stats_b = EvalStats::new();
151
152    // Determine base seed
153    let base_seed = seed.unwrap_or_else(rand::random);
154
155    // Play N hands
156    for hand_num in 0..hands {
157        // Create unique seed for this hand
158        let hand_seed = base_seed.wrapping_add(hand_num as u64);
159
160        // Create and setup engine
161        let mut engine = Engine::new(Some(hand_seed), 1);
162        engine.shuffle();
163        let _ = engine.deal_hand();
164
165        // Record initial stacks
166        let initial_stacks = [engine.players()[0].stack(), engine.players()[1].stack()];
167
168        // Assign AIs to positions (alternate button for fairness)
169        let (ai_0, ai_1, ai_a_position) = if hand_num % 2 == 0 {
170            (ai_policy_a.as_ref(), ai_policy_b.as_ref(), 0)
171        } else {
172            (ai_policy_b.as_ref(), ai_policy_a.as_ref(), 1)
173        };
174
175        // Play hand to completion
176        let (actions, _result_string, showdown, pot) =
177            play_hand_with_two_ais(&mut engine, ai_0, ai_1);
178
179        // Determine winner(s)
180        let (winner_ids, tied) = if let Some(showdown_data) = showdown {
181            if let Some(winners) = showdown_data.get("winners") {
182                if let Some(winners_array) = winners.as_array() {
183                    let winner_vec: Vec<usize> = winners_array
184                        .iter()
185                        .filter_map(|v| v.as_u64().map(|n| n as usize))
186                        .collect();
187                    let is_tie = winner_vec.len() > 1;
188                    (winner_vec, is_tie)
189                } else {
190                    (vec![], false)
191                }
192            } else {
193                (vec![], false)
194            }
195        } else if _result_string.contains("Player 0 wins") {
196            (vec![0], false)
197        } else if _result_string.contains("Player 1 wins") {
198            (vec![1], false)
199        } else {
200            (vec![], false)
201        };
202
203        // Calculate chip deltas
204        let final_stacks = [engine.players()[0].stack(), engine.players()[1].stack()];
205
206        let delta_0 = final_stacks[0] as i64 - initial_stacks[0] as i64;
207        let delta_1 = final_stacks[1] as i64 - initial_stacks[1] as i64;
208
209        // Update statistics based on AI-A's position
210        let (ai_a_won, ai_a_delta) = if ai_a_position == 0 {
211            (winner_ids.contains(&0), delta_0)
212        } else {
213            (winner_ids.contains(&1), delta_1)
214        };
215
216        let ai_b_won = !tied && !ai_a_won;
217        let ai_b_delta = -ai_a_delta;
218
219        // Update action statistics
220        stats_a.update_from_actions(&actions, ai_a_position);
221        stats_b.update_from_actions(&actions, 1 - ai_a_position);
222
223        // Update result statistics
224        stats_a.update_result(ai_a_won, tied, ai_a_delta, pot);
225        stats_b.update_result(ai_b_won, tied, ai_b_delta, pot);
226    }
227
228    // Print results
229    print_eval_results(out, ai_a, ai_b, &stats_a, &stats_b, hands, base_seed)?;
230
231    Ok(())
232}

234/// Play a hand with two AI players
235fn play_hand_with_two_ais(
236    engine: &mut Engine,
237    ai_0: &dyn axiomind_ai::AIOpponent,
238    ai_1: &dyn axiomind_ai::AIOpponent,
239) -> (Vec<ActionRecord>, String, Option<serde_json::Value>, u32) {
240    // Play through the hand
241    while let Ok(current_player) = engine.current_player() {
242        let action = if current_player == 0 {
243            ai_0.get_action(engine, current_player)
244        } else {
245            ai_1.get_action(engine, current_player)
246        };
247
248        match engine.apply_action(current_player, action) {
249            Ok(state) if state.is_hand_complete() => break,
250            Ok(_) => continue,
251            Err(_) => break,
252        }
253    }
254
255    // Get action history
256    let actions = engine.action_history();
257    let pot = engine.pot();
258
259    // Determine winner
260    let (result_string, showdown) = if let Some(folded) = engine.folded_player() {
261        let winner = 1 - folded;
262        (format!("Player {} wins {} (fold)", winner, pot), None)
263    } else if engine.reached_showdown() {
264        let players = engine.players();
265        let board = engine.community_cards();
266
267        let mut player0_cards = Vec::new();
268        let mut player1_cards = Vec::new();
269
270        if let [Some(c1), Some(c2)] = players[0].hole_cards() {
271            player0_cards.push(c1);
272            player0_cards.push(c2);
273        }
274        if let [Some(c1), Some(c2)] = players[1].hole_cards() {
275            player1_cards.push(c1);
276            player1_cards.push(c2);
277        }
278
279        player0_cards.extend_from_slice(&board);
280        player1_cards.extend_from_slice(&board);
281
282        if player0_cards.len() == 7 && player1_cards.len() == 7 {
283            let hand0: [Card; 7] = player0_cards.try_into().unwrap();
284            let hand1: [Card; 7] = player1_cards.try_into().unwrap();
285
286            let strength0 = evaluate_hand(&hand0);
287            let strength1 = evaluate_hand(&hand1);
288
289            let comparison = compare_hands(&strength0, &strength1);
290
291            use std::cmp::Ordering;
292            let (result_str, showdown_info) = match comparison {
293                Ordering::Greater => (
294                    format!("Player 0 wins {} (showdown)", pot),
295                    Some(serde_json::json!({"winners": [0]})),
296                ),
297                Ordering::Less => (
298                    format!("Player 1 wins {} (showdown)", pot),
299                    Some(serde_json::json!({"winners": [1]})),
300                ),
301                Ordering::Equal => (
302                    format!("Split pot {} (tie)", pot),
303                    Some(serde_json::json!({"winners": [0, 1]})),
304                ),
305            };
306            (result_str, showdown_info)
307        } else {
308            ("Unknown result".to_string(), None)
309        }
310    } else {
311        ("Hand incomplete".to_string(), None)
312    };
313
314    (actions, result_string, showdown, pot)
315}

317/// Print evaluation results comparing two AIs
318fn print_eval_results(
319    out: &mut dyn Write,
320    ai_a_name: &str,
321    ai_b_name: &str,
322    stats_a: &EvalStats,
323    stats_b: &EvalStats,
324    hands: u32,
325    seed: u64,
326) -> std::io::Result<()> {
327    let total_hands_a = stats_a.hands_played.max(1) as f64;
328    let total_hands_b = stats_b.hands_played.max(1) as f64;
329
330    writeln!(out, "\nAI Comparison Results")?;
331    writeln!(out, "═══════════════════════════════════════")?;
332    writeln!(out, "Hands played: {}", hands)?;
333    writeln!(out, "Seed: {}", seed)?;
334    writeln!(out)?;
335
336    writeln!(out, "AI-A ({}):", ai_a_name)?;
337    writeln!(out, "  Wins: {} ({:.1}%)", stats_a.wins, stats_a.win_rate())?;
338    writeln!(
339        out,
340        "  Losses: {} ({:.1}%)",
341        stats_a.losses,
342        (stats_a.losses as f64 / total_hands_a) * 100.0
343    )?;
344    writeln!(
345        out,
346        "  Ties: {} ({:.1}%)",
347        stats_a.ties,
348        (stats_a.ties as f64 / total_hands_a) * 100.0
349    )?;
350    writeln!(out, "  Avg chip delta: {:.1}", stats_a.avg_chip_delta())?;
351    writeln!(out, "  Avg pot: {:.1}", stats_a.avg_pot_size())?;
352    writeln!(
353        out,
354        "  Actions: Fold {:.1}% | Check {:.1}% | Call {:.1}% | Bet {:.1}% | Raise {:.1}% | All-in {:.1}%",
355        stats_a.action_percentage(stats_a.folds),
356        stats_a.action_percentage(stats_a.checks),
357        stats_a.action_percentage(stats_a.calls),
358        stats_a.action_percentage(stats_a.bets),
359        stats_a.action_percentage(stats_a.raises),
360        stats_a.action_percentage(stats_a.all_ins),
361    )?;
362    writeln!(out)?;
363
364    writeln!(out, "AI-B ({}):", ai_b_name)?;
365    writeln!(out, "  Wins: {} ({:.1}%)", stats_b.wins, stats_b.win_rate())?;
366    writeln!(
367        out,
368        "  Losses: {} ({:.1}%)",
369        stats_b.losses,
370        (stats_b.losses as f64 / total_hands_b) * 100.0
371    )?;
372    writeln!(
373        out,
374        "  Ties: {} ({:.1}%)",
375        stats_b.ties,
376        (stats_b.ties as f64 / total_hands_b) * 100.0
377    )?;
378    writeln!(out, "  Avg chip delta: {:.1}", stats_b.avg_chip_delta())?;
379    writeln!(out, "  Avg pot: {:.1}", stats_b.avg_pot_size())?;
380    writeln!(
381        out,
382        "  Actions: Fold {:.1}% | Check {:.1}% | Call {:.1}% | Bet {:.1}% | Raise {:.1}% | All-in {:.1}%",
383        stats_b.action_percentage(stats_b.folds),
384        stats_b.action_percentage(stats_b.checks),
385        stats_b.action_percentage(stats_b.calls),
386        stats_b.action_percentage(stats_b.bets),
387        stats_b.action_percentage(stats_b.raises),
388        stats_b.action_percentage(stats_b.all_ins),
389    )?;
390
391    Ok(())
392}
