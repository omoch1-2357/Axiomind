1//! Verify command handler module.
2//!
3//! Validates hand history integrity and game rules compliance for JSONL hand history files.
4//! This module performs comprehensive validation checks including:
5//!
6//! - Board completeness (exactly 5 cards)
7//! - No duplicate cards across board and hole cards
8//! - Chip conservation (net_result must sum to zero)
9//! - Valid hand IDs (format: YYYYMMDD-NNNNNN)
10//! - Betting rules compliance (no illegal reopening after short all-in)
11//! - Player roster consistency across hands
12//! - Street progression validation (Preflop → Flop → Turn → River)
13//! - Stack continuity between hands
14//!
15//! Errors are collected using the shared `BatchValidationError` pattern for structured reporting.
16
17use crate::error::{BatchValidationError, CliError};
18use crate::io_utils::read_text_auto;
19use crate::validation::validate_dealing_meta;
20use std::collections::{HashMap, HashSet};
21use std::io::Write;
22
23/// Type alias for verify-specific batch validation errors.
24/// The `usize` context represents the hand index (1-based) for error reporting.
25type VerifyError = BatchValidationError<usize>;
26
27/// Handle the verify command - validate hand history integrity.
28///
29/// Performs comprehensive validation on JSONL hand history files, checking game rules,
30/// chip conservation laws, and data integrity. Reports all errors found with hand context.
31///
32/// # Arguments
33///
34/// * `input` - Path to JSONL file to verify
35/// * `out` - Output stream for verification results (stdout)
36/// * `err` - Output stream for error messages (stderr)
37///
38/// # Returns
39///
40/// `Result<(), CliError>`: `Ok(())` if all checks pass, otherwise an `Err` that maps to exit code `2`.
41///
42/// # Example
43///
44/// ```no_run
45/// // Verify hand history using CLI
46/// let args = vec!["axiomind", "verify", "--input", "data/hands/session.jsonl"];
47/// let code = axiomind_cli::run(args, &mut std::io::stdout(), &mut std::io::stderr());
48/// assert_eq!(code, 0);
49/// ```
50pub fn handle_verify_command(
51    input: String,
52    out: &mut dyn Write,
53    err: &mut dyn Write,
54) -> Result<(), CliError> {
55    // Enhanced verification with error collection and comprehensive validations
56    let mut errors: Vec<VerifyError> = Vec::new();
57    let mut hands = 0u64;
58    let mut game_over = false;
59    let mut stacks_after_hand: HashMap<String, i64> = HashMap::new();
60    const MIN_CHIP_UNIT: i64 = 25;
61
62    let valid_id = |s: &str| -> bool {
63        s.len() == 15
64            && s[0..8].chars().all(|c| c.is_ascii_digit())
65            && &s[8..9] == "-"
66            && s[9..].chars().all(|c| c.is_ascii_digit())
67    };
68
69    let content = read_text_auto(&input)?;
70
71    for line in content.lines().filter(|l| !l.trim().is_empty()) {
72        hands += 1;
73        if game_over {
74            errors.push(VerifyError {
75                item_context: hands as usize,
76                message: format!(
77                    "Hand {} recorded after player elimination (zero stack)",
78                    hands
79                ),
80            });
81            continue;
82        }
83
84        // Parse as Value first to validate optional net_result chip conservation
85        let v: serde_json::Value = match serde_json::from_str(line) {
86            Ok(v) => v,
87            Err(_) => {
88                errors.push(VerifyError {
89                    item_context: hands as usize,
90                    message: "Invalid JSON record".to_string(),
91                });
92                continue;
93            }
94        };
95
96        // Extract hand_id early for better error reporting
97        let _hand_id = v
98            .get("hand_id")
99            .and_then(|h| h.as_str())
100            .unwrap_or("unknown")
101            .to_string();
102
103        // Level 1 Validation: Check for missing result field
104        let has_result = v.get("result").map(|r| !r.is_null()).unwrap_or(false);
105        if !has_result {
106            errors.push(VerifyError {
107                item_context: hands as usize,
108                message: "Missing or null result field".to_string(),
109            });
110        }
111
112        // Level 2 Validation: Action sequence and street progression
113        if let Some(actions) = v.get("actions").and_then(|a| a.as_array()) {
114            let mut prev_street: Option<String> = None;
115            let streets_order = ["Preflop", "Flop", "Turn", "River"];
116            let mut street_index = 0;
117
118            for action in actions {
119                if let Some(street_str) = action.get("street").and_then(|s| s.as_str()) {
120                    let current_street = street_str.to_string();
121
122                    if let Some(ref prev) = prev_street {
123                        if prev != &current_street {
124                            // Street changed, validate progression
125                            if let Some(current_idx) =
126                                streets_order.iter().position(|s| s == &current_street)
127                            {
128                                if current_idx <= street_index {
129                                    errors.push(VerifyError {
130                                        item_context: hands as usize,
131                                        message: format!(
132                                            "Invalid street progression: {} appears after {}",
133                                            current_street, prev
134                                        ),
135                                    });
136                                } else if current_idx != street_index + 1 {
137                                    errors.push(VerifyError {
138                                        item_context: hands as usize,
139                                        message: format!(
140                                            "Street skipped: jumped from {} to {}",
141                                            prev, current_street
142                                        ),
143                                    });
144                                }
145                                street_index = current_idx;
146                            }
147                        }
148                    } else {
149                        // First street
150                        if let Some(idx) = streets_order.iter().position(|s| s == &current_street) {
151                            street_index = idx;
152                        }
153                    }
154                    prev_street = Some(current_street);
155                }
156            }
157        }
158
159        let mut starting_stacks: Option<HashMap<String, i64>> = None;
160        if let Some(players) = v.get("players").and_then(|p| p.as_array()) {
161            let mut start_map = HashMap::new();
162            for player in players {
163                let Some(id) = player.get("id").and_then(|x| x.as_str()) else {
164                    continue;
165                };
166                let stack = player
167                    .get("stack_start")
168                    .and_then(|x| x.as_i64())
169                    .unwrap_or(0);
170                start_map.insert(id.to_string(), stack);
171            }
172            let prev_state = if stacks_after_hand.is_empty() {
173                None
174            } else {
175                Some(&stacks_after_hand)
176            };
177
178            // Collect roster state errors
179            if let Some(prev_map) = prev_state {
180                for (id, stack_start) in &start_map {
181                    if let Some(prev_stack) = prev_map.get(id) {
182                        if *prev_stack != *stack_start {
183                            errors.push(VerifyError {
184                                item_context: hands as usize,
185                                message: format!("Stack mismatch for {} at hand {}", id, hands),
186                            });
187                        }
188                        if *prev_stack <= 0 {
189                            errors.push(VerifyError {
190                                item_context: hands as usize,
191                                message: format!("Player {} reappeared after elimination", id),
192                            });
193                        }
194                    } else {
195                        errors.push(VerifyError {
196                            item_context: hands as usize,
197                            message: format!("Unexpected player {} at hand {}", id, hands),
198                        });
199                    }
200                }
201                for (id, prev_stack) in prev_map {
202                    if !start_map.contains_key(id) && *prev_stack > 0 {
203                        errors.push(VerifyError {
204                            item_context: hands as usize,
205                            message: format!("Missing player {} at hand {}", id, hands),
206                        });
207                    }
208                }
209            }
210            for (id, stack_start) in &start_map {
211                if *stack_start <= 0 {
212                    errors.push(VerifyError {
213                        item_context: hands as usize,
214                        message: format!("Player {} has non-positive starting stack", id),
215                    });
216                }
217            }
218
219            starting_stacks = Some(start_map.clone());
220            stacks_after_hand = start_map;
221            if let Some(nr_obj) = v.get("net_result").and_then(|x| x.as_object()) {
222                for id in nr_obj.keys() {
223                    if !stacks_after_hand.contains_key(id) {
224                        errors.push(VerifyError {
225                            item_context: hands as usize,
226                            message: format!("Unknown player {} in net_result", id),
227                        });
228                    }
229                }
230            }
231        }
232
233        if let (Some(start_map), Some(meta_obj)) = (
234            starting_stacks.as_ref(),
235            v.get("meta").and_then(|m| m.as_object()),
236        ) {
237            let button_id = v.get("button").and_then(|b| b.as_str());
238            if let Err(msg) = validate_dealing_meta(meta_obj, button_id, start_map, hands) {
239                errors.push(VerifyError {
240                    item_context: hands as usize,
241                    message: msg,
242                });
243            }
244        }
245
246        let mut big_blind = MIN_CHIP_UNIT;
247        if let Some(blinds_val) = v.get("blinds") {
248            if let Some(bb) = blinds_val.get("bb").and_then(|x| x.as_i64()) {
249                big_blind = bb;
250            } else if let Some(arr) = blinds_val.as_array()
251                && arr.len() >= 2
252                && let Some(bb) = arr[1].as_i64()
253            {
254                big_blind = bb;
255            }
256        }
257        if big_blind < MIN_CHIP_UNIT {
258            big_blind = MIN_CHIP_UNIT;
259        }
260
261        if let Some(actions) = v.get("actions").and_then(|a| a.as_array())
262            && let Some(ref start_map) = starting_stacks
263            && let Err(msg) = ensure_no_reopen_after_short_all_in(
264                actions,
265                big_blind,
266                MIN_CHIP_UNIT,
267                start_map,
268                hands,
269            )
270        {
271            errors.push(VerifyError {
272                item_context: hands as usize,
273                message: msg,
274            });
275        }
276
277        if let Some(nr) = v.get("net_result").and_then(|x| x.as_object()) {
278            let mut sum: i64 = 0;
279            for val in nr.values() {
280                if let Some(n) = val.as_i64() {
281                    sum += n;
282                }
283            }
284            if sum != 0 {
285                errors.push(VerifyError {
286                    item_context: hands as usize,
287                    message: "Chip conservation violated".to_string(),
288                });
289            }
290            for (id, delta) in nr.iter() {
291                if let Some(val) = delta.as_i64() {
292                    let entry = stacks_after_hand.entry(id.clone()).or_insert(0);
293                    *entry += val;
294                }
295            }
296            if stacks_after_hand.values().any(|stack| *stack <= 0) {
297                game_over = true;
298            }
299        }
300
301        match serde_json::from_value::<axiomind_engine::logger::HandRecord>(v.clone()) {
302            Ok(rec) => {
303                if rec.board.len() != 5 {
304                    errors.push(VerifyError {
305                        item_context: hands as usize,
306                        message: format!(
307                            "Invalid board length: expected 5 cards but found {}",
308                            rec.board.len()
309                        ),
310                    });
311                }
312
313                let mut seen_cards: HashSet<axiomind_engine::cards::Card> = HashSet::new();
314                let mut duplicate_cards: HashSet<axiomind_engine::cards::Card> = HashSet::new();
315                {
316                    let mut record_card = |card: axiomind_engine::cards::Card| {
317                        if !seen_cards.insert(card) {
318                            duplicate_cards.insert(card);
319                        }
320                    };
321                    for card in &rec.board {
322                        record_card(*card);
323                    }
324                    if let Some(players) = v.get("players").and_then(|p| p.as_array()) {
325                        for player in players {
326                            let pid = player
327                                .get("id")
328                                .and_then(|x| x.as_str())
329                                .unwrap_or("unknown");
330                            if let Some(hole_cards) =
331                                player.get("hole_cards").and_then(|h| h.as_array())
332                            {
333                                for card_val in hole_cards {
334                                    match serde_json::from_value::<axiomind_engine::cards::Card>(
335                                        card_val.clone(),
336                                    ) {
337                                        Ok(card) => record_card(card),
338                                        Err(_) => {
339                                            errors.push(VerifyError {
340                                                item_context: hands as usize,
341                                                message: format!(
342                                                    "Invalid card specification for {}",
343                                                    pid
344                                                ),
345                                            });
346                                        }
347                                    }
348                                }
349                            }
350                        }
351                    }
352                }
353
354                if !duplicate_cards.is_empty() {
355                    let mut cards: Vec<String> = duplicate_cards
356                        .iter()
357                        .map(|card| format!("{:?} {:?}", card.rank, card.suit))
358                        .collect();
359                    cards.sort();
360                    errors.push(VerifyError {
361                        item_context: hands as usize,
362                        message: format!("Duplicate card(s) detected: {}", cards.join(", ")),
363                    });
364                }
365
366                if !valid_id(&rec.hand_id) {
367                    errors.push(VerifyError {
368                        item_context: hands as usize,
369                        message: "Invalid hand_id format".to_string(),
370                    });
371                }
372            }
373            Err(_) => {
374                errors.push(VerifyError {
375                    item_context: hands as usize,
376                    message: "Invalid record structure".to_string(),
377                });
378            }
379        }
380    }
381
382    // Output results
383    if errors.is_empty() {
384        writeln!(out, "Verify: OK (hands={})", hands)?;
385        Ok(())
386    } else {
387        writeln!(out, "Verify: FAIL (hands={})", hands)?;
388        writeln!(err)?;
389        writeln!(err, "Errors found:")?;
390        for error in &errors {
391            writeln!(err, "  Hand {}: {}", error.item_context, error.message)?;
392        }
393        writeln!(err)?;
394        let invalid_hand_numbers: HashSet<usize> = errors.iter().map(|e| e.item_context).collect();
395        let invalid_hands = invalid_hand_numbers.len() as u64;
396        let percentage = if hands > 0 {
397            (invalid_hands as f64 / hands as f64 * 100.0).round() as u32
398        } else {
399            0
400        };
401        writeln!(
402            err,
403            "Summary: {} error(s) in {} hands ({} invalid hands, {}% invalid)",
404            errors.len(),
405            hands,
406            invalid_hands,
407            percentage
408        )?;
409        Err(CliError::InvalidInput(format!(
410            "{} validation errors found",
411            errors.len()
412        )))
413    }
414}

416/// Module-private helper: Validate that no illegal reopening occurs after a short all-in.
417///
418/// In No-Limit Hold'em, a short all-in (less than a full raise) does not reopen betting
419/// for players who have already acted. This function checks action sequences to ensure
420/// this rule is enforced.
421fn ensure_no_reopen_after_short_all_in(
422    actions: &[serde_json::Value],
423    big_blind: i64,
424    min_chip_unit: i64,
425    starting_stacks: &HashMap<String, i64>,
426    hand_index: u64,
427) -> Result<(), String> {
428    #[derive(Clone, Copy)]
429    enum ActionKind {
430        Bet(i64),
431        Raise(i64),
432        AllIn(Option<i64>),
433        Call,
434        Check,
435        Fold,
436        Other,
437    }
438
439    let mut remaining = starting_stacks.clone();
440    let mut prev_street: Option<String> = None;
441    let mut street_committed: HashMap<String, i64> = HashMap::new();
442    let mut current_high: i64 = 0;
443    let mut last_full_raise: i64 = big_blind.max(min_chip_unit);
444    let mut reopen_blocked = false;
445
446    let extract_player_id = |act: &serde_json::Value| -> Option<String> {
447        act.get("player_id")
448            .and_then(|v| v.as_str().map(|s| s.to_string()))
449            .or_else(|| {
450                act.get("player_id").and_then(|v| {
451                    v.as_i64().map(|n| {
452                        let candidate = format!("p{}", n);
453                        if starting_stacks.contains_key(&candidate) {
454                            candidate
455                        } else {
456                            n.to_string()
457                        }
458                    })
459                })
460            })
461    };
462
463    for (idx, act) in actions.iter().enumerate() {
464        let Some(player_id) = extract_player_id(act) else {
465            continue;
466        };
467
468        if !starting_stacks.contains_key(&player_id) {
469            return Err(format!(
470                "Unknown player {} at hand {} (action #{})",
471                player_id,
472                hand_index,
473                idx + 1
474            ));
475        }
476
477        if let Some(street) = act.get("street").and_then(|s| s.as_str())
478            && prev_street.as_deref() != Some(street)
479        {
480            prev_street = Some(street.to_string());
481            street_committed.clear();
482            current_high = 0;
483            last_full_raise = big_blind.max(min_chip_unit);
484            reopen_blocked = false;
485        }
486
487        let action_kind: ActionKind = match act.get("action") {
488            Some(serde_json::Value::Object(map)) => {
489                if let Some(amount) = map.get("Bet").and_then(|v| v.as_i64()) {
490                    ActionKind::Bet(amount)
491                } else if let Some(amount) = map.get("Raise").and_then(|v| v.as_i64()) {
492                    ActionKind::Raise(amount)
493                } else if let Some(amount) = map.get("AllIn").and_then(|v| v.as_i64()) {
494                    ActionKind::AllIn(Some(amount))
495                } else if map.get("Call").is_some() {
496                    ActionKind::Call
497                } else if map.get("Check").is_some() {
498                    ActionKind::Check
499                } else if map.get("Fold").is_some() {
500                    ActionKind::Fold
501                } else {
502                    ActionKind::Other
503                }
504            }
505            Some(serde_json::Value::String(name)) => match name.to_ascii_lowercase().as_str() {
506                "bet" => {
507                    return Err(format!(
508                        "Bet action missing amount at hand {} (action #{})",
509                        hand_index,
510                        idx + 1
511                    ));
512                }
513                "raise" => {
514                    return Err(format!(
515                        "Bet action missing amount at hand {} (action #{})",
516                        hand_index,
517                        idx + 1
518                    ));
519                }
520                "call" => ActionKind::Call,
521                "check" => ActionKind::Check,
522                "fold" => ActionKind::Fold,
523                "allin" | "all-in" => ActionKind::AllIn(None),
524                _ => ActionKind::Other,
525            },
526            _ => ActionKind::Other,
527        };
528
529        let commit_before = street_committed.get(&player_id).copied().unwrap_or(0);
530        let mut target_commit = commit_before;
531
532        match action_kind {
533            ActionKind::Bet(amount) => {
534                if amount % min_chip_unit != 0 {
535                    return Err(format!(
536                        "Invalid bet amount {} at hand {} (action #{})",
537                        amount,
538                        hand_index,
539                        idx + 1
540                    ));
541                }
542                let min_bet = big_blind.max(min_chip_unit);
543                if amount < min_bet {
544                    return Err(format!(
545                        "Bet below minimum {} at hand {} (action #{})",
546                        min_bet,
547                        hand_index,
548                        idx + 1
549                    ));
550                }
551                target_commit = amount;
552            }
553            ActionKind::Raise(amount) => {
554                if amount % min_chip_unit != 0 {
555                    return Err(format!(
556                        "Invalid raise amount {} at hand {} (action #{})",
557                        amount,
558                        hand_index,
559                        idx + 1
560                    ));
561                }
562                let min_delta = last_full_raise.max(big_blind).max(min_chip_unit);
563                if amount < min_delta {
564                    return Err(format!(
565                        "Raise delta {} below minimum {} at hand {} (action #{})",
566                        amount,
567                        min_delta,
568                        hand_index,
569                        idx + 1
570                    ));
571                }
572                target_commit = current_high + amount;
573            }
574            ActionKind::AllIn(Some(amount)) => {
575                if amount % min_chip_unit != 0 {
576                    return Err(format!(
577                        "Invalid all-in amount {} at hand {} (action #{})",
578                        amount,
579                        hand_index,
580                        idx + 1
581                    ));
582                }
583                target_commit = commit_before + amount;
584            }
585            ActionKind::AllIn(None) => {
586                let remaining_stack = remaining.get(&player_id).copied().unwrap_or(0);
587                if remaining_stack % min_chip_unit != 0 {
588                    return Err(format!(
589                        "Invalid all-in amount {} at hand {} (action #{})",
590                        remaining_stack,
591                        hand_index,
592                        idx + 1
593                    ));
594                }
595                target_commit = commit_before + remaining_stack;
596            }
597            ActionKind::Call => {
598                target_commit = current_high.max(commit_before);
599            }
600            ActionKind::Check | ActionKind::Fold | ActionKind::Other => {}
601        }
602
603        let mut delta_chips = target_commit.saturating_sub(commit_before);
604        if let Some(rem) = remaining.get(&player_id)
605            && delta_chips > *rem
606        {
607            delta_chips = *rem;
608        }
609        let new_commit = commit_before + delta_chips;
610
611        if delta_chips > 0
612            && let Some(rem) = remaining.get_mut(&player_id)
613        {
614            if *rem < delta_chips {
615                return Err(format!(
616                    "Player {} commits more chips than stack at hand {} (action #{})",
617                    player_id,
618                    hand_index,
619                    idx + 1
620                ));
621            }
622            *rem -= delta_chips;
623        }
624
625        let extra = new_commit.saturating_sub(current_high);
626        if extra > 0 {
627            if reopen_blocked {
628                return Err(format!(
629                    "Betting illegally reopened after short all-in at hand {} (action #{})",
630                    hand_index,
631                    idx + 1
632                ));
633            }
634            let min_full_raise = last_full_raise.max(big_blind).max(min_chip_unit);
635            if extra < min_full_raise {
636                reopen_blocked = true;
637            } else {
638                last_full_raise = extra;
639                reopen_blocked = false;
640            }
641            current_high = new_commit;
642        } else {
643            current_high = current_high.max(new_commit);
644        }
645
646        street_committed.insert(player_id, new_commit);
647    }
648
649    Ok(())
650}

736    #[test]
737    fn test_valid_hand_id_format() {
738        let valid_id = |s: &str| -> bool {
739            s.len() == 15
740                && s[0..8].chars().all(|c| c.is_ascii_digit())
741                && &s[8..9] == "-"
742                && s[9..].chars().all(|c| c.is_ascii_digit())
743        };
744
745        assert!(valid_id("20250101-000001"));
746        assert!(valid_id("19700101-123456"));
747        assert!(!valid_id("2025-01-01-000001")); // Wrong format
748        assert!(!valid_id("20250101000001")); // Missing dash
749        assert!(!valid_id("20250101-abcdef")); // Non-digits
750    }
