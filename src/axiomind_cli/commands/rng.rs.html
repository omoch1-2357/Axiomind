1//! Random number generator verification command.
2//!
3//! The `rng` command verifies the properties of the ChaCha20 random number generator
4//! used by the poker engine. It generates sample random values and displays them
5//! for inspection.
6//!
7//! ## Purpose
8//!
9//! This command is primarily used for:
10//! - Verifying RNG determinism (same seed produces same sequence)
11//! - Debugging random number generation issues
12//! - Validating RNG distribution properties
13
14use crate::error::CliError;
15use rand::{RngCore, SeedableRng};
16use std::io::Write;
17
18/// Handle the rng command - verify random number generator properties.
19///
20/// Generates and displays a sample of random numbers using the ChaCha20 RNG
21/// with the specified seed (or a random seed if not provided).
22///
23/// # Arguments
24///
25/// * `seed` - Optional seed value for the RNG (uses random seed if None)
26/// * `out` - Output stream for RNG sample values
27///
28/// # Returns
29///
30/// * `Ok(())` on success
31/// * `Err(CliError)` on write failure
32///
33/// # Example
34///
35/// ```ignore
36/// # use axiomind_cli::commands::handle_rng_command;
37/// # use std::io;
38/// let mut out = io::stdout();
39/// handle_rng_command(Some(12345), &mut out).expect("RNG command failed");
40/// ```
41pub fn handle_rng_command(seed: Option<u64>, out: &mut dyn Write) -> Result<(), CliError> {
42    let s = seed.unwrap_or_else(rand::random);
43    let mut rng = rand_chacha::ChaCha20Rng::seed_from_u64(s);
44    let mut vals = vec![];
45    for _ in 0..5 {
46        vals.push(rng.next_u64());
47    }
48    writeln!(out, "RNG sample: {:?}", vals)?;
49    Ok(())
50}
