1//! Dataset splitting command handler for ML training preparation.
2//!
3//! This module provides functionality to split JSONL hand history files into
4//! train/validation/test sets for machine learning workflows. Supports both
5//! in-memory and streaming modes for handling large datasets efficiently.
6//!
7//! # Environment Variables
8//!
9//! - `AXIOMIND_DATASET_STREAM_THRESHOLD`: Minimum records for streaming mode (default: 10000)
10//! - `AXIOMIND_DATASET_STREAM_TRACE`: Enable streaming debug output
11//!
12//! # Output Files
13//!
14//! Creates `train.jsonl`, `val.jsonl`, and `test.jsonl` in the output directory.
15//!
16//! # Examples
17//!
18//! ```no_run
19//! // Split dataset using CLI
20//! let args = vec![
21//!     "axiomind", "dataset",
22//!     "--input", "data/sim.jsonl",
23//!     "--outdir", "data/splits",
24//!     "--train", "0.7",
25//!     "--val", "0.2",
26//!     "--test", "0.1",
27//!     "--seed", "42"
28//! ];
29//! let code = axiomind_cli::run(args, &mut std::io::stdout(), &mut std::io::stderr());
30//! assert_eq!(code, 0);
31//! ```
32

40/// Handle the dataset command: create training/validation/test splits.
41///
42/// Splits a JSONL hand history file into train/val/test sets for machine learning.
43/// Supports both in-memory and streaming modes for large datasets.
44///
45/// # Arguments
46///
47/// * `input` - Path to input JSONL file
48/// * `output_dir` - Output directory for split files
49/// * `train` - Training set proportion (0.0-1.0 or percentage)
50/// * `val` - Validation set proportion (0.0-1.0 or percentage)
51/// * `test` - Test set proportion (0.0-1.0 or percentage)
52/// * `seed` - RNG seed for reproducible shuffling
53/// * `out` - Output stream for normal messages
54/// * `err` - Output stream for error messages
55///
56/// # Returns
57///
58/// `Ok(())` on success, or `CliError` on failure
59///
60/// # Environment Variables
61///
62/// - `AXIOMIND_DATASET_STREAM_THRESHOLD`: Min records for streaming mode (default: 10000)
63/// - `AXIOMIND_DATASET_STREAM_TRACE`: Enable streaming debug output
64#[allow(clippy::too_many_arguments)]
65pub fn handle_dataset_command(
66    input: String,
67    output_dir: String,
68    train: Option<f64>,
69    val: Option<f64>,
70    test: Option<f64>,
71    seed: Option<u64>,
72    _out: &mut dyn Write,
73    err: &mut dyn Write,
74) -> Result<(), CliError> {
75    // Try streaming mode first
76    match dataset_stream_if_needed(&input, &output_dir, train, val, test, seed, err)? {
77        Some(()) => return Ok(()),
78        None => { /* Continue with normal in-memory processing */ }
79    }
80
81    // In-memory processing for smaller datasets
82    let content = std::fs::read_to_string(&input).map_err(|e| {
83        let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
84        CliError::Io(e)
85    })?;
86
87    let mut lines: Vec<String> = content
88        .lines()
89        .filter(|l| !l.trim().is_empty())
90        .map(|s| s.to_string())
91        .collect();
92
93    let n = lines.len();
94    if n == 0 {
95        ui::write_error(err, "Empty input")?;
96        return Err(CliError::InvalidInput("Empty input".to_string()));
97    }
98
99    let splits = compute_splits(train, val, test).map_err(|msg| {
100        let _ = ui::write_error(err, &msg);
101        CliError::InvalidInput(msg)
102    })?;
103
104    let tr = splits[0];
105    let va = splits[1];
106    let te = splits[2];
107    let sum = tr + va + te;
108
109    if (sum - 1.0).abs() > 1e-6 {
110        ui::write_error(err, "Splits must sum to 100% (1.0 total)")?;
111        return Err(CliError::InvalidInput(
112            "Splits must sum to 100% (1.0 total)".to_string(),
113        ));
114    }
115
116    let mut rng = ChaCha20Rng::seed_from_u64(seed.unwrap_or(0));
117    lines.shuffle(&mut rng);
118
119    let n_tr = ((tr * n as f64).round() as usize).min(n);
120    let n_va = ((va * n as f64).round() as usize).min(n.saturating_sub(n_tr));
121    let _n_te = n.saturating_sub(n_tr + n_va);
122
123    // Validate all records
124    for (idx, raw) in lines.iter().enumerate() {
125        let trimmed = raw.trim();
126        if let Err(e) = serde_json::from_str::<axiomind_engine::logger::HandRecord>(trimmed) {
127            ui::write_error(err, &format!("Invalid record at line {}: {}", idx + 1, e))?;
128            return Err(CliError::InvalidInput(format!(
129                "Invalid record at line {}: {}",
130                idx + 1,
131                e
132            )));
133        }
134    }
135
136    let (trv, rest) = lines.split_at(n_tr);
137    let (vav, tev) = rest.split_at(n_va);
138
139    let out_root = std::path::Path::new(&output_dir);
140    std::fs::create_dir_all(out_root).map_err(|e| {
141        let _ = ui::write_error(
142            err,
143            &format!("Failed to create directory {}: {}", output_dir, e),
144        );
145        CliError::Io(e)
146    })?;
147
148    let mut write_split = |name: &str, data: &[String]| -> Result<(), CliError> {
149        let path = out_root.join(name);
150        let file = std::fs::File::create(&path).map_err(|e| {
151            let _ = ui::write_error(err, &format!("Failed to create {}: {}", path.display(), e));
152            CliError::Io(e)
153        })?;
154        let mut writer = BufWriter::new(file);
155        for l in data {
156            writeln!(writer, "{}", l).map_err(|e| {
157                let _ = ui::write_error(err, &format!("Failed to write {}: {}", path.display(), e));
158                CliError::Io(e)
159            })?;
160        }
161        Ok(())
162    };
163
164    write_split("train.jsonl", trv)?;
165    write_split("val.jsonl", vav)?;
166    write_split("test.jsonl", tev)?;
167
168    Ok(())
169}

171/// Compute dataset split ratios from optional inputs.
172///
173/// Handles both percentage (>1.0) and ratio (0.0-1.0) formats.
174/// Defaults to 80% train, 10% val, 10% test if not specified.
175///
176/// # Arguments
177///
178/// * `train` - Optional train ratio/percentage
179/// * `val` - Optional validation ratio/percentage
180/// * `test` - Optional test ratio/percentage
181///
182/// # Returns
183///
184/// Array of [train_ratio, val_ratio, test_ratio] or error message
185fn compute_splits(
186    train: Option<f64>,
187    val: Option<f64>,
188    test: Option<f64>,
189) -> Result<[f64; 3], String> {
190    const DEFAULTS: [f64; 3] = [0.8, 0.1, 0.1];
191    let mut splits = [0.0; 3];
192
193    for (idx, opt) in [train, val, test].into_iter().enumerate() {
194        splits[idx] = match opt {
195            Some(v) if v.is_sign_negative() => {
196                return Err("Splits must be non-negative".into());
197            }
198            Some(v) if v > 1.0 + 1e-6 => v / 100.0,
199            Some(v) => v,
200            None => DEFAULTS[idx],
201        };
202    }
203
204    let sum: f64 = splits.iter().sum();
205    if (sum - 1.0).abs() > 1e-6 {
206        return Err("Splits must sum to 100% (1.0 total)".into());
207    }
208
209    Ok(splits)
210}

212/// Attempt streaming mode for large datasets.
213///
214/// If the dataset exceeds the threshold, use streaming mode to avoid loading
215/// everything into memory. Returns Some(()) if streaming was used, None otherwise.
216fn dataset_stream_if_needed(
217    input: &str,
218    outdir: &str,
219    train: Option<f64>,
220    val: Option<f64>,
221    test: Option<f64>,
222    seed: Option<u64>,
223    err: &mut dyn Write,
224) -> Result<Option<()>, CliError> {
225    fn strip_utf8_bom(s: &mut String) {
226        const UTF8_BOM: &str = "\u{feff}";
227        if s.starts_with(UTF8_BOM) {
228            s.drain(..UTF8_BOM.len());
229        }
230    }
231
232    let threshold = std::env::var("AXIOMIND_DATASET_STREAM_THRESHOLD")
233        .ok()
234        .and_then(|v| v.parse::<usize>().ok())
235        .unwrap_or(10_000);
236
237    if threshold == 0 {
238        return Ok(None);
239    }
240
241    let trace_stream = std::env::var("AXIOMIND_DATASET_STREAM_TRACE")
242        .map(|v| {
243            matches!(
244                v.trim().to_ascii_lowercase().as_str(),
245                "1" | "true" | "yes" | "on"
246            )
247        })
248        .unwrap_or(false);
249
250    let count_file = std::fs::File::open(input).map_err(|e| {
251        let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
252        CliError::Io(e)
253    })?;
254
255    let mut record_count = 0usize;
256    {
257        let reader = BufReader::new(count_file);
258        let mut first_line = true;
259        for line in reader.lines() {
260            let mut line = line.map_err(|e| {
261                let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
262                CliError::Io(e)
263            })?;
264            if first_line {
265                strip_utf8_bom(&mut line);
266                first_line = false;
267            }
268            if !line.trim().is_empty() {
269                record_count += 1;
270            }
271        }
272    }
273
274    if record_count == 0 {
275        ui::write_error(err, "Empty input")?;
276        return Err(CliError::InvalidInput("Empty input".to_string()));
277    }
278
279    if record_count <= threshold {
280        return Ok(None);
281    }
282
283    let splits = compute_splits(train, val, test).map_err(|msg| {
284        let _ = ui::write_error(err, &msg);
285        CliError::InvalidInput(msg)
286    })?;
287
288    let tr = splits[0];
289    let va = splits[1];
290    let n = record_count;
291    let n_tr = ((tr * n as f64).round() as usize).min(n);
292    let n_va = ((va * n as f64).round() as usize).min(n.saturating_sub(n_tr));
293
294    let mut rng = ChaCha20Rng::seed_from_u64(seed.unwrap_or(0));
295    let mut indices: Vec<usize> = (0..record_count).collect();
296    indices.shuffle(&mut rng);
297
298    #[derive(Clone, Copy)]
299    enum SplitSlot {
300        Train,
301        Val,
302        Test,
303    }
304
305    let mut assignments = vec![SplitSlot::Test; record_count];
306    for &idx in indices.iter().take(n_tr) {
307        assignments[idx] = SplitSlot::Train;
308    }
309    for &idx in indices.iter().skip(n_tr).take(n_va) {
310        assignments[idx] = SplitSlot::Val;
311    }
312
313    std::fs::create_dir_all(outdir).map_err(|e| {
314        let _ = ui::write_error(
315            err,
316            &format!("Failed to create directory {}: {}", outdir, e),
317        );
318        CliError::Io(e)
319    })?;
320
321    if trace_stream {
322        ui::write_error(
323            err,
324            &format!("Streaming dataset input (records={})", record_count),
325        )?;
326    }
327
328    let out_root = std::path::Path::new(outdir);
329    let train_path = out_root.join("train.jsonl");
330    let val_path = out_root.join("val.jsonl");
331    let test_path = out_root.join("test.jsonl");
332
333    let mut create_writer = |path: &std::path::Path| -> Result<BufWriter<std::fs::File>, CliError> {
334        std::fs::File::create(path)
335            .map(BufWriter::new)
336            .map_err(|e| {
337                let _ =
338                    ui::write_error(err, &format!("Failed to create {}: {}", path.display(), e));
339                CliError::Io(e)
340            })
341    };
342
343    let mut train_writer = create_writer(&train_path)?;
344    let mut val_writer = create_writer(&val_path)?;
345    let mut test_writer = create_writer(&test_path)?;
346
347    let data_file = std::fs::File::open(input).map_err(|e| {
348        let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
349        CliError::Io(e)
350    })?;
351
352    let reader = BufReader::new(data_file);
353    let mut record_idx = 0usize;
354    let mut first_line = true;
355
356    for (line_idx, line_res) in reader.lines().enumerate() {
357        let mut line = line_res.map_err(|e| {
358            let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
359            CliError::Io(e)
360        })?;
361
362        if first_line {
363            strip_utf8_bom(&mut line);
364            first_line = false;
365        }
366
367        if line.trim().is_empty() {
368            continue;
369        }
370
371        if let Err(e) = serde_json::from_str::<axiomind_engine::logger::HandRecord>(&line) {
372            ui::write_error(
373                err,
374                &format!("Invalid record at line {}: {}", line_idx + 1, e),
375            )?;
376            return Err(CliError::InvalidInput(format!(
377                "Invalid record at line {}: {}",
378                line_idx + 1,
379                e
380            )));
381        }
382
383        let bucket = assignments
384            .get(record_idx)
385            .copied()
386            .unwrap_or(SplitSlot::Test);
387        record_idx += 1;
388
389        match bucket {
390            SplitSlot::Train => {
391                writeln!(train_writer, "{}", line)?;
392            }
393            SplitSlot::Val => {
394                writeln!(val_writer, "{}", line)?;
395            }
396            SplitSlot::Test => {
397                writeln!(test_writer, "{}", line)?;
398            }
399        }
400    }
401
402    Ok(Some(()))
403}
