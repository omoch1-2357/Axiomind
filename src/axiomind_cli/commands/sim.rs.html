1//! Simulation command handler for large-scale hand generation.
2//!
3//! This module provides functionality to run large-scale poker hand simulations,
4//! generating specified numbers of hands with configurable parameters. It supports
5//! environment variables for fast mode and breaking early for testing purposes.
6//!
7//! # Environment Variables
8//!
9//! - `AXIOMIND_SIM_FAST`: Enable fast simulation mode (batch writes, minimal output)
10//! - `AXIOMIND_SIM_BREAK_AFTER`: Break after N hands (for testing)
11//! - `AXIOMIND_SIM_SLEEP_MICROS`: Delay between hands in microseconds
12//!
13//! # Examples
14//!
15//! ```no_run
16//! // Run simulation using CLI
17//! let args = vec![
18//!     "axiomind", "sim",
19//!     "--hands", "1000",
20//!     "--output", "data/sim.jsonl",
21//!     "--seed", "42",
22//!     "--level", "1"
23//! ];
24//! let code = axiomind_cli::run(args, &mut std::io::stdout(), &mut std::io::stderr());
25//! assert_eq!(code, 0);
26//! ```
27

35/// Handle the sim command: run large-scale hand simulations.
36///
37/// Generates and optionally records N hands of poker. Supports resuming from
38/// previous runs and breaking early for testing via environment variables.
39///
40/// # Arguments
41///
42/// * `hands` - Total number of hands to simulate
43/// * `output` - Path to save hand histories (JSONL format)
44/// * `seed` - Base RNG seed (each hand uses seed + hand_index)
45/// * `level` - Blind level (1-20)
46/// * `resume` - Resume from existing JSONL file (skips completed hands)
47/// * `out` - Output stream for normal messages
48/// * `err` - Output stream for error messages
49///
50/// # Returns
51///
52/// `Ok(())` on success, or `CliError` on failure
53///
54/// # Environment Variables
55///
56/// - `AXIOMIND_SIM_FAST`: Enable fast mode (batch writes, minimal output)
57/// - `AXIOMIND_SIM_BREAK_AFTER`: Break after N hands (for testing)
58/// - `AXIOMIND_SIM_SLEEP_MICROS`: Delay between hands in microseconds
59pub fn handle_sim_command(
60    hands: u64,
61    output: Option<String>,
62    seed: Option<u64>,
63    level: Option<u8>,
64    resume: Option<String>,
65    out: &mut dyn Write,
66    err: &mut dyn Write,
67) -> Result<(), CliError> {
68    let total: usize = hands as usize;
69    if total == 0 {
70        ui::write_error(err, "hands must be >= 1")?;
71        return Err(CliError::InvalidInput("hands must be >= 1".to_string()));
72    }
73
74    let level = level.unwrap_or(1).clamp(1, 20);
75    let mut completed = 0usize;
76    let mut path = None;
77
78    if let Some(outp) = output.clone() {
79        path = Some(std::path::PathBuf::from(outp));
80    }
81
82    // Resume: count existing unique hand_ids and warn on duplicates
83    if let Some(res) = resume.as_ref() {
84        let contents = std::fs::read_to_string(res).unwrap_or_default();
85        let mut seen = std::collections::HashSet::new();
86        let mut dups = 0usize;
87
88        for line in contents.lines().filter(|l| !l.trim().is_empty()) {
89            let hid = serde_json::from_str::<serde_json::Value>(line)
90                .ok()
91                .and_then(|v| {
92                    v.get("hand_id")
93                        .and_then(|x| x.as_str())
94                        .map(|s| s.to_string())
95                })
96                .unwrap_or_default();
97            if hid.is_empty() {
98                continue;
99            }
100            if !seen.insert(hid) {
101                dups += 1;
102            }
103        }
104
105        completed = seen.len();
106        path = Some(std::path::PathBuf::from(res));
107
108        if dups > 0 {
109            writeln!(err, "Warning: {} duplicate hand_id(s) skipped", dups)?;
110        }
111        writeln!(out, "Resumed from {}", completed)?;
112    }
113
114    let base_seed = seed.unwrap_or_else(rand::random);
115    let mut eng = Engine::new(Some(base_seed), level);
116    eng.shuffle();
117
118    let break_after = std::env::var("AXIOMIND_SIM_BREAK_AFTER")
119        .ok()
120        .and_then(|v| v.parse::<usize>().ok());
121    let per_hand_delay = std::env::var("AXIOMIND_SIM_SLEEP_MICROS")
122        .ok()
123        .and_then(|v| v.parse::<u64>().ok())
124        .map(std::time::Duration::from_micros);
125    let fast_mode = std::env::var("AXIOMIND_SIM_FAST")
126        .map(|v| {
127            matches!(
128                v.trim().to_ascii_lowercase().as_str(),
129                "1" | "true" | "yes" | "on"
130            )
131        })
132        .unwrap_or(false);
133
134    if !fast_mode {
135        let _ = &per_hand_delay;
136    }
137
138    if fast_mode {
139        return sim_run_fast(
140            total,
141            level,
142            seed,
143            base_seed,
144            break_after,
145            per_hand_delay,
146            completed,
147            path.as_deref(),
148            out,
149            err,
150        );
151    }
152
153    #[allow(clippy::mut_range_bound)]
154    for i in completed..total {
155        // Create a fresh engine per hand to avoid residual hole cards
156        let mut e = Engine::new(Some(base_seed + i as u64), level);
157        e.shuffle();
158        let _ = e.deal_hand();
159
160        // Play the hand to completion
161        let (actions, result, showdown) = play_hand_to_completion(&mut e);
162
163        if let Some(p) = &path {
164            if let Err(e) = ensure_parent_dir(p) {
165                ui::write_error(err, &e)?;
166                return Err(CliError::Io(std::io::Error::other(e)));
167            }
168
169            let mut f = match std::fs::OpenOptions::new()
170                .create(true)
171                .append(true)
172                .open(p)
173            {
174                Ok(file) => file,
175                Err(e) => {
176                    ui::write_error(err, &format!("Failed to open output file: {}", e))?;
177                    return Err(CliError::Io(e));
178                }
179            };
180
181            let hand_id = format!("19700101-{:06}", i + 1);
182            let board = e.board().clone();
183            let rec = serde_json::json!({
184                "hand_id": hand_id,
185                "seed": base_seed + i as u64,
186                "level": level,
187                "actions": actions,
188                "board": board,
189                "result": result,
190                "ts": "1970-01-01T00:00:00+00:00".to_string(),
191                "meta": null,
192                "showdown": showdown
193            });
194
195            let json_str = match serde_json::to_string(&rec) {
196                Ok(s) => s,
197                Err(e) => {
198                    ui::write_error(err, &format!("Failed to serialize hand: {}", e))?;
199                    return Err(CliError::InvalidInput(format!(
200                        "Failed to serialize hand: {}",
201                        e
202                    )));
203                }
204            };
205
206            if writeln!(f, "{}", json_str).is_err() {
207                ui::write_error(err, "Failed to write hand to file")?;
208                return Err(CliError::Io(std::io::Error::other(
209                    "Failed to write hand to file",
210                )));
211            }
212        }
213
214        completed += 1;
215
216        if let Some(b) = break_after
217            && completed == b
218        {
219            writeln!(out, "Interrupted: saved {}/{}", completed, total)?;
220            return Err(CliError::Interrupted(format!(
221                "Interrupted: saved {}/{}",
222                completed, total
223            )));
224        }
225    }
226
227    writeln!(out, "Simulated: {} hands", completed)?;
228    Ok(())
229}

231/// Play a hand to completion using baseline AI for both players.
232///
233/// This module-private helper function simulates a complete poker hand by having
234/// both players use the baseline AI strategy until the hand reaches completion.
235///
236/// # Arguments
237///
238/// * `engine` - Mutable reference to the game engine with dealt cards
239///
240/// # Returns
241///
242/// A tuple containing:
243/// - Action history (Vec of ActionRecords)
244/// - Result string describing the outcome
245/// - Optional showdown information (JSON value with winners)
246fn play_hand_to_completion(
247    engine: &mut Engine,
248) -> (
249    Vec<axiomind_engine::logger::ActionRecord>,
250    String,
251    Option<serde_json::Value>,
252) {
253    use axiomind_engine::cards::Card;
254    use axiomind_engine::hand::{compare_hands, evaluate_hand};
255
256    let ai = create_ai("baseline");
257
258    // Play through the hand
259    while let Ok(current_player) = engine.current_player() {
260        let action = ai.get_action(engine, current_player);
261
262        match engine.apply_action(current_player, action) {
263            Ok(state) if state.is_hand_complete() => break,
264            Ok(_) => continue,
265            Err(_) => break,
266        }
267    }
268
269    // Get action history
270    let actions = engine.action_history();
271
272    // Determine winner
273    let (result_string, showdown) = if let Some(folded) = engine.folded_player() {
274        // Someone folded - other player wins
275        let winner = 1 - folded;
276        let pot = engine.pot();
277        (format!("Player {} wins {} (fold)", winner, pot), None)
278    } else if engine.reached_showdown() {
279        // Evaluate hands at showdown
280        let players = engine.players();
281        let board = engine.community_cards();
282
283        // Build 7-card hands for each player
284        let mut player0_cards = Vec::new();
285        let mut player1_cards = Vec::new();
286
287        if let [Some(c1), Some(c2)] = players[0].hole_cards() {
288            player0_cards.push(c1);
289            player0_cards.push(c2);
290        }
291        if let [Some(c1), Some(c2)] = players[1].hole_cards() {
292            player1_cards.push(c1);
293            player1_cards.push(c2);
294        }
295
296        player0_cards.extend_from_slice(&board);
297        player1_cards.extend_from_slice(&board);
298
299        if player0_cards.len() == 7 && player1_cards.len() == 7 {
300            let hand0: [Card; 7] = player0_cards.try_into().unwrap();
301            let hand1: [Card; 7] = player1_cards.try_into().unwrap();
302
303            let strength0 = evaluate_hand(&hand0);
304            let strength1 = evaluate_hand(&hand1);
305
306            let pot = engine.pot();
307            let comparison = compare_hands(&strength0, &strength1);
308
309            use std::cmp::Ordering;
310            let (result_str, showdown_info) = match comparison {
311                Ordering::Greater => (
312                    format!("Player 0 wins {} (showdown)", pot),
313                    Some(serde_json::json!({"winners": [0]})),
314                ),
315                Ordering::Less => (
316                    format!("Player 1 wins {} (showdown)", pot),
317                    Some(serde_json::json!({"winners": [1]})),
318                ),
319                Ordering::Equal => (
320                    format!("Split pot {} (tie)", pot),
321                    Some(serde_json::json!({"winners": [0, 1]})),
322                ),
323            };
324            (result_str, showdown_info)
325        } else {
326            ("Unknown result".to_string(), None)
327        }
328    } else {
329        ("No result".to_string(), None)
330    };
331
332    (actions, result_string, showdown)
333}

335/// Run simulation in fast mode with batch writes.
336///
337/// This module-private helper function optimizes simulation performance by
338/// using buffered writes and reducing output overhead.
339///
340/// # Arguments
341///
342/// * `total` - Total number of hands to simulate
343/// * `level` - Blind level (1-20)
344/// * `_seed` - Original seed parameter (unused, kept for signature compatibility)
345/// * `base_seed` - Base RNG seed for hand generation
346/// * `break_after` - Optional break point for early termination
347/// * `per_hand_delay` - Optional delay between hands
348/// * `completed` - Number of hands already completed (from resume)
349/// * `path` - Optional path for output file
350/// * `out` - Output stream for normal messages
351/// * `err` - Output stream for error messages
352///
353/// # Returns
354///
355/// `Ok(())` on success, or `CliError` on failure
356#[allow(clippy::too_many_arguments)]
357fn sim_run_fast(
358    total: usize,
359    level: u8,
360    _seed: Option<u64>,
361    base_seed: u64,
362    break_after: Option<usize>,
363    per_hand_delay: Option<std::time::Duration>,
364    mut completed: usize,
365    path: Option<&std::path::Path>,
366    out: &mut dyn Write,
367    err: &mut dyn Write,
368) -> Result<(), CliError> {
369    let mut writer = match path {
370        Some(p) => {
371            if let Err(e) = ensure_parent_dir(p) {
372                ui::write_error(err, &e)?;
373                return Err(CliError::Io(std::io::Error::other(e)));
374            }
375
376            match std::fs::OpenOptions::new()
377                .create(true)
378                .append(true)
379                .open(p)
380            {
381                Ok(file) => Some(std::io::BufWriter::new(file)),
382                Err(e) => {
383                    ui::write_error(err, &format!("Failed to open {}: {}", p.display(), e))?;
384                    return Err(CliError::Io(e));
385                }
386            }
387        }
388        None => None,
389    };
390
391    #[allow(clippy::mut_range_bound)]
392    for i in completed..total {
393        let mut engine = Engine::new(Some(base_seed + i as u64), level);
394        engine.shuffle();
395        let _ = engine.deal_hand();
396
397        // Play the hand to completion
398        let (actions, result, showdown) = play_hand_to_completion(&mut engine);
399
400        if let Some(w) = writer.as_mut() {
401            let hand_id = format!("19700101-{:06}", i + 1);
402            let board = engine.board().clone();
403            let record = serde_json::json!({
404                "hand_id": hand_id,
405                "seed": base_seed + i as u64,
406                "level": level,
407                "actions": actions,
408                "board": board,
409                "result": result,
410                "ts": "1970-01-01T00:00:00+00:00".to_string(),
411                "meta": null,
412                "showdown": showdown
413            });
414            if let Err(e) = writeln!(w, "{}", serde_json::to_string(&record).unwrap()) {
415                ui::write_error(err, "Failed to write simulation output")?;
416                return Err(CliError::Io(e));
417            }
418        }
419
420        completed += 1;
421
422        if let Some(delay) = per_hand_delay {
423            std::thread::sleep(delay);
424        }
425
426        if let Some(b) = break_after
427            && completed == b
428        {
429            if let Some(w) = writer.as_mut()
430                && let Err(e) = w.flush()
431            {
432                ui::write_error(err, "Failed to flush simulation output")?;
433                return Err(CliError::Io(e));
434            }
435            writeln!(out, "Interrupted: saved {}/{}", completed, total)?;
436            return Err(CliError::Interrupted(format!(
437                "Interrupted: saved {}/{}",
438                completed, total
439            )));
440        }
441    }
442
443    if let Some(mut w) = writer
444        && let Err(e) = w.flush()
445    {
446        ui::write_error(err, "Failed to flush simulation output")?;
447        return Err(CliError::Io(e));
448    }
449
450    writeln!(out, "Simulated: {} hands", completed)?;
451    Ok(())
452}
