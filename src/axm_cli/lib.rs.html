1//! # Axiomind CLI Library
2//!
3//! This library provides the command-line interface for the Axiomind poker engine.
4//! It exposes subcommands for playing, simulating, analyzing, and verifying poker hands.
5//!
6//! ## Main Entry Point
7//!
8//! The primary entry point is the [`run`] function, which parses command-line arguments
9//! and executes the appropriate subcommand.
10//!
11//! ## Example Usage
12//!
13//! ```no_run
14//! use std::io;
15//! let args = vec!["axm", "play", "--vs", "ai", "--hands", "10"];
16//! let code = axm_cli::run(args, &mut io::stdout(), &mut io::stderr());
17//! assert_eq!(code, 0);
18//! ```
19//!
20//! ## Available Subcommands
21//!
22//! - `play`: Play poker hands against AI or human opponents
23//! - `sim`: Run large-scale simulations and generate hand histories
24//! - `stats`: Aggregate statistics from JSONL hand history files
25//! - `verify`: Validate game rules and hand history integrity
26//! - `replay`: Replay previously recorded hands
27//! - `deal`: Deal a single hand for inspection
28//! - `bench`: Benchmark hand evaluation performance
29//! - `eval`: Evaluate AI policies head-to-head
30//! - `export`: Convert hand histories to various formats (CSV, JSON, SQLite)
31//! - `dataset`: Create training/validation/test splits for ML
32//! - `cfg`: Display current configuration settings
33//! - `doctor`: Run environment diagnostics
34//! - `rng`: Verify RNG properties

47/// Reads a line of input from a buffered reader, blocking until available.
48/// Returns None on EOF or read error. Returns Some with trimmed content (which may be empty).
49fn read_stdin_line(stdin: &mut dyn BufRead) -> Option<String> {
50    let mut line = String::new();
51    match stdin.read_line(&mut line) {
52        Ok(0) => None, // EOF
53        Ok(_) => {
54            let trimmed = line.trim();
55            Some(trimmed.to_string())
56        }
57        Err(_) => None, // Read error
58    }
59}

61/// Result type for parsing user input into player actions
62#[derive(Debug, PartialEq)]
63enum ParseResult {
64    /// Valid player action
65    Action(axm_engine::player::PlayerAction),
66    /// User wants to quit
67    Quit,
68    /// Invalid input with error message
69    Invalid(String),
70}

72/// Format a PlayerAction as a human-readable string
73fn format_action(action: &axm_engine::player::PlayerAction) -> String {
74    match action {
75        axm_engine::player::PlayerAction::Fold => "fold".to_string(),
76        axm_engine::player::PlayerAction::Check => "check".to_string(),
77        axm_engine::player::PlayerAction::Call => "call".to_string(),
78        axm_engine::player::PlayerAction::Bet(amount) => format!("bet {}", amount),
79        axm_engine::player::PlayerAction::Raise(amount) => format!("raise {}", amount),
80        axm_engine::player::PlayerAction::AllIn => "all-in".to_string(),
81    }
82}

84/// Parse user input string into a PlayerAction
85/// Returns ParseResult indicating success, quit, or error
86fn parse_player_action(input: &str) -> ParseResult {
87    let input = input.trim().to_lowercase();
88    let parts: Vec<&str> = input.split_whitespace().collect();
89
90    if parts.is_empty() {
91        return ParseResult::Invalid("Empty input".to_string());
92    }
93
94    // Check for quit commands first
95    if parts[0] == "q" || parts[0] == "quit" {
96        return ParseResult::Quit;
97    }
98
99    // Parse actions
100    match parts[0] {
101        "fold" => ParseResult::Action(axm_engine::player::PlayerAction::Fold),
102        "check" => ParseResult::Action(axm_engine::player::PlayerAction::Check),
103        "call" => ParseResult::Action(axm_engine::player::PlayerAction::Call),
104        "allin" | "all-in" => ParseResult::Action(axm_engine::player::PlayerAction::AllIn),
105        "bet" => {
106            if parts.len() < 2 {
107                return ParseResult::Invalid("Bet requires an amount (e.g., 'bet 100')".to_string());
108            }
109            match parts[1].parse::<u32>() {
110                Ok(amount) if amount > 0 => {
111                    ParseResult::Action(axm_engine::player::PlayerAction::Bet(amount))
112                }
113                Ok(_) => ParseResult::Invalid("Bet amount must be positive".to_string()),
114                Err(_) => ParseResult::Invalid("Invalid bet amount".to_string()),
115            }
116        }
117        "raise" => {
118            if parts.len() < 2 {
119                return ParseResult::Invalid(
120                    "Raise requires an amount (e.g., 'raise 50')".to_string(),
121                );
122            }
123            match parts[1].parse::<u32>() {
124                Ok(amount) if amount > 0 => {
125                    ParseResult::Action(axm_engine::player::PlayerAction::Raise(amount))
126                }
127                Ok(_) => ParseResult::Invalid("Raise amount must be positive".to_string()),
128                Err(_) => ParseResult::Invalid("Invalid raise amount".to_string()),
129            }
130        }
131        _ => ParseResult::Invalid(format!(
132            "Unrecognized action '{}'. Valid actions: fold, check, call, bet <amount>, raise <amount>, allin, q",
133            parts[0]
134        )),
135    }
136}

138/// Execute the play command with specified parameters
139/// Returns exit code (0 = success, non-zero = error)
140fn execute_play_command(
141    vs: Vs,
142    hands: u32,
143    seed: Option<u64>,
144    level: u8,
145    stdin: &mut dyn BufRead,
146    out: &mut dyn Write,
147    err: &mut dyn Write,
148) -> i32 {
149    if hands == 0 {
150        let _ = ui::write_error(err, "hands must be >= 1");
151        return 2;
152    }
153
154    let seed = seed.unwrap_or_else(rand::random);
155    let level = level.max(1);
156
157    // Display warning for AI placeholder mode
158    if matches!(vs, Vs::Ai) {
159        let _ = ui::display_warning(
160            err,
161            "AI opponent is a placeholder that always checks. Use for demo purposes only.",
162        );
163    }
164
165    let _ = writeln!(
166        out,
167        "play: vs={} hands={} seed={}",
168        vs.as_str(),
169        hands,
170        seed
171    );
172    let _ = writeln!(out, "Level: {}", level);
173
174    let mut eng = Engine::new(Some(seed), level);
175    eng.shuffle();
176
177    let mut played = 0u32;
178    let mut quit_requested = false;
179
180    for i in 1..=hands {
181        if quit_requested {
182            break;
183        }
184
185        // level progression: +1 every 15 hands
186        let cur_level: u8 =
187            level.saturating_add(((i - 1) / axm_engine::player::HANDS_PER_LEVEL) as u8);
188        if i > 1 {
189            let _ = writeln!(out, "Level: {}", cur_level);
190        }
191        eng.set_level(cur_level);
192        let (sb, bb) = eng.blinds();
193        let _ = writeln!(out, "Blinds: SB={} BB={}", sb, bb);
194        let _ = writeln!(out, "Hand {}", i);
195        if let Err(e) = eng.deal_hand() {
196            let _ = ui::write_error(err, &format!("Failed to deal hand: {}", e));
197            return 2;
198        }
199
200        match vs {
201            Vs::Human => {
202                let human_player_id = 0;
203
204                loop {
205                    // エンジンから現在のアクターを取得
206                    let current_player = eng.current_player();
207
208                    if current_player == human_player_id {
209                        // 人間のターン
210                        let _ = write!(out, "Enter action (check/call/bet/raise/fold/q): ");
211                        let _ = out.flush();
212
213                        match read_stdin_line(stdin) {
214                            Some(input) => match parse_player_action(&input) {
215                                ParseResult::Action(action) => {
216                                    match eng.apply_action(human_player_id, action.clone()) {
217                                        Ok(state) => {
218                                            let action_str = format_action(&action);
219                                            let _ = writeln!(out, "Action: {}", action_str);
220                                            let _ = writeln!(out, "Pot: {}", state.pot());
221                                            if state.is_hand_complete() {
222                                                let _ = writeln!(out, "Hand complete.");
223                                                break;
224                                            }
225                                        }
226                                        Err(e) => {
227                                            let _ = ui::write_error(
228                                                err,
229                                                &format!("Invalid action: {}", e),
230                                            );
231                                        }
232                                    }
233                                }
234                                ParseResult::Quit => {
235                                    quit_requested = true;
236                                    break;
237                                }
238                                ParseResult::Invalid(msg) => {
239                                    let _ = ui::write_error(err, &msg);
240                                }
241                            },
242                            None => {
243                                quit_requested = true;
244                                break;
245                            }
246                        }
247                    } else {
248                        let mut ai_action = axm_engine::player::PlayerAction::Check;
249                        match eng.apply_action(current_player, ai_action.clone()) {
250                            Ok(state) => {
251                                let _ = writeln!(out, "AI: {}", format_action(&ai_action));
252                                let _ = writeln!(out, "Pot: {}", state.pot());
253                                if state.is_hand_complete() {
254                                    let _ = writeln!(out, "Hand complete.");
255                                    break;
256                                }
257                            }
258                            Err(err_check) => {
259                                let err_check_msg = err_check.to_string();
260                                ai_action = axm_engine::player::PlayerAction::Call;
261                                match eng.apply_action(current_player, ai_action.clone()) {
262                                    Ok(state) => {
263                                        let _ = writeln!(out, "AI: {}", format_action(&ai_action));
264                                        let _ = writeln!(out, "Pot: {}", state.pot());
265                                        if state.is_hand_complete() {
266                                            let _ = writeln!(out, "Hand complete.");
267                                            break;
268                                        }
269                                    }
270                                    Err(err_call) => {
271                                        let _ = ui::write_error(
272                                            err,
273                                            &format!(
274                                                "AI action failed: check -> {}; call -> {}",
275                                                err_check_msg, err_call
276                                            ),
277                                        );
278                                        break;
279                                    }
280                                }
281                            }
282                        }
283                    }
284                }
285            }
286            Vs::Ai => {
287                // 既存の AI モードプレースホルダー
288                let _ = writeln!(out, "{}", ui::tag_demo_output("ai: check"));
289            }
290        }
291        played += 1;
292    }
293
294    let _ = writeln!(out, "Session hands={}", hands);
295    let _ = writeln!(out, "Hands played: {} (completed)", played);
296    0
297}

299fn ensure_no_reopen_after_short_all_in(
300    actions: &[serde_json::Value],
301    big_blind: i64,
302    min_chip_unit: i64,
303    starting_stacks: &HashMap<String, i64>,
304    hand_index: u64,
305) -> Result<(), String> {
306    #[derive(Clone, Copy)]
307    enum ActionKind {
308        Bet(i64),
309        Raise(i64),
310        AllIn(Option<i64>),
311        Call,
312        Check,
313        Fold,
314        Other,
315    }
316
317    let mut remaining = starting_stacks.clone();
318    let mut prev_street: Option<String> = None;
319    let mut street_committed: HashMap<String, i64> = HashMap::new();
320    let mut current_high: i64 = 0;
321    let mut last_full_raise: i64 = big_blind.max(min_chip_unit);
322    let mut reopen_blocked = false;
323
324    let extract_player_id = |act: &serde_json::Value| -> Option<String> {
325        act.get("player_id")
326            .and_then(|v| v.as_str().map(|s| s.to_string()))
327            .or_else(|| {
328                act.get("player_id").and_then(|v| {
329                    v.as_i64().map(|n| {
330                        let candidate = format!("p{}", n);
331                        if starting_stacks.contains_key(&candidate) {
332                            candidate
333                        } else {
334                            n.to_string()
335                        }
336                    })
337                })
338            })
339    };
340
341    for (idx, act) in actions.iter().enumerate() {
342        let Some(player_id) = extract_player_id(act) else {
343            continue;
344        };
345
346        if !starting_stacks.contains_key(&player_id) {
347            return Err(format!(
348                "Unknown player {} at hand {} (action #{})",
349                player_id,
350                hand_index,
351                idx + 1
352            ));
353        }
354
355        if let Some(street) = act.get("street").and_then(|s| s.as_str()) {
356            if prev_street.as_deref() != Some(street) {
357                prev_street = Some(street.to_string());
358                street_committed.clear();
359                current_high = 0;
360                last_full_raise = big_blind.max(min_chip_unit);
361                reopen_blocked = false;
362            }
363        }
364
365        let action_kind: ActionKind = match act.get("action") {
366            Some(serde_json::Value::Object(map)) => {
367                if let Some(amount) = map.get("Bet").and_then(|v| v.as_i64()) {
368                    ActionKind::Bet(amount)
369                } else if let Some(amount) = map.get("Raise").and_then(|v| v.as_i64()) {
370                    ActionKind::Raise(amount)
371                } else if let Some(amount) = map.get("AllIn").and_then(|v| v.as_i64()) {
372                    ActionKind::AllIn(Some(amount))
373                } else if map.get("Call").is_some() {
374                    ActionKind::Call
375                } else if map.get("Check").is_some() {
376                    ActionKind::Check
377                } else if map.get("Fold").is_some() {
378                    ActionKind::Fold
379                } else {
380                    ActionKind::Other
381                }
382            }
383            Some(serde_json::Value::String(name)) => match name.to_ascii_lowercase().as_str() {
384                "bet" => {
385                    return Err(format!(
386                        "Bet action missing amount at hand {} (action #{})",
387                        hand_index,
388                        idx + 1
389                    ))
390                }
391                "raise" => {
392                    return Err(format!(
393                        "Bet action missing amount at hand {} (action #{})",
394                        hand_index,
395                        idx + 1
396                    ))
397                }
398                "call" => ActionKind::Call,
399                "check" => ActionKind::Check,
400                "fold" => ActionKind::Fold,
401                "allin" | "all-in" => ActionKind::AllIn(None),
402                _ => ActionKind::Other,
403            },
404            _ => ActionKind::Other,
405        };
406
407        let commit_before = street_committed.get(&player_id).copied().unwrap_or(0);
408        let mut target_commit = commit_before;
409
410        match action_kind {
411            ActionKind::Bet(amount) => {
412                if amount % min_chip_unit != 0 {
413                    return Err(format!(
414                        "Invalid bet amount {} at hand {} (action #{})",
415                        amount,
416                        hand_index,
417                        idx + 1
418                    ));
419                }
420                let min_bet = big_blind.max(min_chip_unit);
421                if amount < min_bet {
422                    return Err(format!(
423                        "Bet below minimum {} at hand {} (action #{})",
424                        min_bet,
425                        hand_index,
426                        idx + 1
427                    ));
428                }
429                target_commit = amount;
430            }
431            ActionKind::Raise(amount) => {
432                if amount % min_chip_unit != 0 {
433                    return Err(format!(
434                        "Invalid raise amount {} at hand {} (action #{})",
435                        amount,
436                        hand_index,
437                        idx + 1
438                    ));
439                }
440                let min_delta = last_full_raise.max(big_blind).max(min_chip_unit);
441                if amount < min_delta {
442                    return Err(format!(
443                        "Raise delta {} below minimum {} at hand {} (action #{})",
444                        amount,
445                        min_delta,
446                        hand_index,
447                        idx + 1
448                    ));
449                }
450                target_commit = current_high + amount;
451            }
452            ActionKind::AllIn(Some(amount)) => {
453                if amount % min_chip_unit != 0 {
454                    return Err(format!(
455                        "Invalid all-in amount {} at hand {} (action #{})",
456                        amount,
457                        hand_index,
458                        idx + 1
459                    ));
460                }
461                target_commit = commit_before + amount;
462            }
463            ActionKind::AllIn(None) => {
464                let remaining_stack = remaining.get(&player_id).copied().unwrap_or(0);
465                if remaining_stack % min_chip_unit != 0 {
466                    return Err(format!(
467                        "Invalid all-in amount {} at hand {} (action #{})",
468                        remaining_stack,
469                        hand_index,
470                        idx + 1
471                    ));
472                }
473                target_commit = commit_before + remaining_stack;
474            }
475            ActionKind::Call => {
476                target_commit = current_high.max(commit_before);
477            }
478            ActionKind::Check | ActionKind::Fold | ActionKind::Other => {}
479        }
480
481        let mut delta_chips = target_commit.saturating_sub(commit_before);
482        if let Some(rem) = remaining.get(&player_id) {
483            if delta_chips > *rem {
484                delta_chips = *rem;
485            }
486        }
487        let new_commit = commit_before + delta_chips;
488
489        if delta_chips > 0 {
490            if let Some(rem) = remaining.get_mut(&player_id) {
491                if *rem < delta_chips {
492                    return Err(format!(
493                        "Player {} commits more chips than stack at hand {} (action #{})",
494                        player_id,
495                        hand_index,
496                        idx + 1
497                    ));
498                }
499                *rem -= delta_chips;
500            }
501        }
502
503        let extra = new_commit.saturating_sub(current_high);
504        if extra > 0 {
505            if reopen_blocked {
506                return Err(format!(
507                    "Betting illegally reopened after short all-in at hand {} (action #{})",
508                    hand_index,
509                    idx + 1
510                ));
511            }
512            let min_full_raise = last_full_raise.max(big_blind).max(min_chip_unit);
513            if extra < min_full_raise {
514                reopen_blocked = true;
515            } else {
516                last_full_raise = extra;
517                reopen_blocked = false;
518            }
519            current_high = new_commit;
520        } else {
521            current_high = current_high.max(new_commit);
522        }
523
524        street_committed.insert(player_id, new_commit);
525    }
526
527    Ok(())
528}

530fn validate_dealing_meta(
531    meta: &serde_json::Map<String, serde_json::Value>,
532    button: Option<&str>,
533    starting_stacks: &HashMap<String, i64>,
534    hand_index: u64,
535) -> Result<(), String> {
536    if starting_stacks.is_empty() {
537        return Ok(());
538    }
539    let player_count = starting_stacks.len();
540    let rounds = 2; // Texas Hold'em: two hole cards per player
541    let sb = meta.get("small_blind").and_then(|v| v.as_str());
542    let bb = meta.get("big_blind").and_then(|v| v.as_str());
543    if let Some(sb_id) = sb {
544        if !starting_stacks.contains_key(sb_id) {
545            return Err(format!(
546                "Invalid dealing order at hand {}: unknown small blind {}",
547                hand_index, sb_id
548            ));
549        }
550    }
551    if let Some(bb_id) = bb {
552        if !starting_stacks.contains_key(bb_id) {
553            return Err(format!(
554                "Invalid dealing order at hand {}: unknown big blind {}",
555                hand_index, bb_id
556            ));
557        }
558    }
559    if let (Some(btn), Some(sb_id)) = (button, sb) {
560        if sb_id != btn {
561            return Err(format!(
562                "Invalid dealing order at hand {}: button {} must match small blind {}",
563                hand_index, btn, sb_id
564            ));
565        }
566    }
567    if let (Some(sb_id), Some(bb_id)) = (sb, bb) {
568        if sb_id == bb_id {
569            return Err(format!(
570                "Invalid dealing order at hand {}: small blind and big blind must differ",
571                hand_index
572            ));
573        }
574        if player_count == 2 {
575            if let Some(expected_bb) = starting_stacks
576                .keys()
577                .find(|id| id.as_str() != sb_id)
578                .map(|s| s.as_str())
579            {
580                if bb_id != expected_bb {
581                    return Err(format!(
582                        "Invalid dealing order at hand {}: expected big blind {} but found {}",
583                        hand_index, expected_bb, bb_id
584                    ));
585                }
586            }
587        }
588    }
589    if let Some(seq_val) = meta.get("deal_sequence") {
590        let seq = seq_val.as_array().ok_or_else(|| {
591            format!(
592                "Invalid dealing order at hand {}: deal_sequence must be an array",
593                hand_index
594            )
595        })?;
596        let seq_ids: Option<Vec<&str>> = seq.iter().map(|v| v.as_str()).collect();
597        let seq_ids = seq_ids.ok_or_else(|| {
598            format!(
599                "Invalid dealing order at hand {}: deal_sequence must contain player identifiers",
600                hand_index
601            )
602        })?;
603        let expected_len = player_count * rounds;
604        if seq_ids.len() != expected_len {
605            return Err(format!(
606                "Invalid dealing order at hand {}: expected {} entries in deal_sequence but found {}",
607                hand_index,
608                expected_len,
609                seq_ids.len()
610            ));
611        }
612        let known: HashSet<&str> = starting_stacks.keys().map(|k| k.as_str()).collect();
613        if seq_ids.iter().any(|id| !known.contains(id)) {
614            return Err(format!(
615                "Invalid dealing order at hand {}: deal_sequence references unknown player",
616                hand_index
617            ));
618        }
619        let first_round = &seq_ids[..player_count];
620        if let Some(sb_id) = sb {
621            if first_round.first().copied() != Some(sb_id) {
622                return Err(format!(
623                    "Invalid dealing order at hand {}: expected {} to receive the first card",
624                    hand_index, sb_id
625                ));
626            }
627        }
628        if let Some(bb_id) = bb {
629            if player_count >= 2 && first_round.get(1).copied() != Some(bb_id) {
630                return Err(format!(
631                    "Invalid dealing order at hand {}: expected {} to receive the second card",
632                    hand_index, bb_id
633                ));
634            }
635        }
636        let first_round_set: HashSet<&str> = first_round.iter().copied().collect();
637        if first_round_set.len() != player_count {
638            return Err(format!(
639                "Invalid dealing order at hand {}: duplicate players in first deal round",
640                hand_index
641            ));
642        }
643        for round_idx in 1..rounds {
644            let chunk = &seq_ids[round_idx * player_count..(round_idx + 1) * player_count];
645            if chunk != first_round {
646                return Err(format!(
647                    "Invalid dealing order at hand {}: inconsistent card distribution order",
648                    hand_index
649                ));
650            }
651        }
652    }
653    if let Some(burn_val) = meta.get("burn_positions") {
654        let burn_arr = burn_val.as_array().ok_or_else(|| {
655            format!(
656                "Invalid dealing order at hand {}: burn_positions must be an array",
657                hand_index
658            )
659        })?;
660        let burn_positions: Option<Vec<i64>> = burn_arr.iter().map(|v| v.as_i64()).collect();
661        let burn_positions = burn_positions.ok_or_else(|| {
662            format!(
663                "Invalid dealing order at hand {}: burn_positions must contain integers",
664                hand_index
665            )
666        })?;
667        if burn_positions.len() != 3 {
668            return Err(format!(
669                "Invalid dealing order at hand {}: expected 3 burn positions",
670                hand_index
671            ));
672        }
673        let player_count_i64 = player_count as i64;
674        if player_count_i64 >= 2 {
675            let hole_cards = player_count_i64 * 2;
676            let expected = vec![
677                hole_cards + 1,
678                hole_cards + 1 + 3 + 1,
679                hole_cards + 1 + 3 + 1 + 1 + 1,
680            ];
681            if burn_positions != expected {
682                return Err(format!(
683                    "Invalid dealing order at hand {}: expected burn positions {:?} but found {:?}",
684                    hand_index, expected, burn_positions
685                ));
686            }
687        }
688    }
689    Ok(())
690}

692fn validate_roster_state(
693    prev: Option<&HashMap<String, i64>>,
694    current: &HashMap<String, i64>,
695    hands: u64,
696    err: &mut dyn Write,
697    ok: &mut bool,
698) {
699    if let Some(prev_map) = prev {
700        for (id, stack_start) in current {
701            if let Some(prev_stack) = prev_map.get(id) {
702                if *prev_stack != *stack_start {
703                    *ok = false;
704                    let _ = ui::write_error(
705                        err,
706                        &format!("Stack mismatch for {} at hand {}", id, hands),
707                    );
708                }
709                if *prev_stack <= 0 {
710                    *ok = false;
711                    let _ = ui::write_error(
712                        err,
713                        &format!(
714                            "Player {} reappeared after elimination at hand {}",
715                            id, hands
716                        ),
717                    );
718                }
719            } else {
720                *ok = false;
721                let _ =
722                    ui::write_error(err, &format!("Unexpected player {} at hand {}", id, hands));
723            }
724        }
725        for (id, prev_stack) in prev_map {
726            if !current.contains_key(id) && *prev_stack > 0 {
727                *ok = false;
728                let _ = ui::write_error(err, &format!("Missing player {} at hand {}", id, hands));
729            }
730        }
731    }
732    for (id, stack_start) in current {
733        if *stack_start <= 0 {
734            *ok = false;
735            let _ = ui::write_error(
736                err,
737                &format!(
738                    "Player {} has non-positive starting stack at hand {}",
739                    id, hands
740                ),
741            );
742        }
743    }
744}

796    fn read_text_auto(path: &str) -> Result<String, String> {
797        let mut content = if path.ends_with(".zst") {
798            // Read entire compressed file then decompress; more portable across platforms
799            let comp = std::fs::read(path).map_err(|e| e.to_string())?;
800            // Use a conservative initial capacity; zstd will grow as needed
801            let dec = zstd::bulk::decompress(&comp, 8 * 1024 * 1024).map_err(|e| e.to_string())?;
802            String::from_utf8(dec).map_err(|e| e.to_string())?
803        } else {
804            std::fs::read_to_string(path).map_err(|e| e.to_string())?
805        };
806        strip_utf8_bom(&mut content);
807        Ok(content)
808    }

818    fn compute_splits(
819        train: Option<f64>,
820        val: Option<f64>,
821        test: Option<f64>,
822    ) -> Result<[f64; 3], String> {
823        const DEFAULTS: [f64; 3] = [0.8, 0.1, 0.1];
824        let mut splits = [0.0; 3];
825        for (idx, opt) in [train, val, test].into_iter().enumerate() {
826            splits[idx] = match opt {
827                Some(v) if v.is_sign_negative() => {
828                    return Err("Splits must be non-negative".into());
829                }
830                Some(v) if v > 1.0 + 1e-6 => v / 100.0,
831                Some(v) => v,
832                None => DEFAULTS[idx],
833            };
834        }
835        let sum: f64 = splits.iter().sum();
836        if (sum - 1.0).abs() > 1e-6 {
837            return Err("Splits must sum to 100% (1.0 total)".into());
838        }
839        Ok(splits)
840    }

842    fn dataset_stream_if_needed(
843        input: &str,
844        outdir: &str,
845        train: Option<f64>,
846        val: Option<f64>,
847        test: Option<f64>,
848        seed: Option<u64>,
849        err: &mut dyn Write,
850    ) -> Option<i32> {
851        use std::io::{BufRead, BufReader, BufWriter};
852
853        let threshold = std::env::var("AXM_DATASET_STREAM_THRESHOLD")
854            .ok()
855            .and_then(|v| v.parse::<usize>().ok())
856            .unwrap_or(10_000);
857        if threshold == 0 {
858            return None;
859        }
860
861        let trace_stream = std::env::var("AXM_DATASET_STREAM_TRACE")
862            .map(|v| {
863                matches!(
864                    v.trim().to_ascii_lowercase().as_str(),
865                    "1" | "true" | "yes" | "on"
866                )
867            })
868            .unwrap_or(false);
869
870        let count_file = match std::fs::File::open(input) {
871            Ok(f) => f,
872            Err(e) => {
873                let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
874                return Some(2);
875            }
876        };
877
878        let mut record_count = 0usize;
879        {
880            let reader = BufReader::new(count_file);
881            let mut first_line = true;
882            for line in reader.lines() {
883                match line {
884                    Ok(mut line) => {
885                        if first_line {
886                            strip_utf8_bom(&mut line);
887                            first_line = false;
888                        }
889                        if !line.trim().is_empty() {
890                            record_count += 1;
891                        }
892                    }
893                    Err(e) => {
894                        let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
895                        return Some(2);
896                    }
897                }
898            }
899        }
900
901        if record_count == 0 {
902            let _ = ui::write_error(err, "Empty input");
903            return Some(2);
904        }
905
906        if record_count <= threshold {
907            return None;
908        }
909
910        let splits = match compute_splits(train, val, test) {
911            Ok(v) => v,
912            Err(msg) => {
913                let _ = ui::write_error(err, &msg);
914                return Some(2);
915            }
916        };
917
918        let tr = splits[0];
919        let va = splits[1];
920        let n = record_count;
921        let n_tr = ((tr * n as f64).round() as usize).min(n);
922        let n_va = ((va * n as f64).round() as usize).min(n.saturating_sub(n_tr));
923
924        let mut rng = rand_chacha::ChaCha20Rng::seed_from_u64(seed.unwrap_or(0));
925        let mut indices: Vec<usize> = (0..record_count).collect();
926        indices.shuffle(&mut rng);
927
928        #[derive(Clone, Copy)]
929        enum SplitSlot {
930            Train,
931            Val,
932            Test,
933        }
934
935        let mut assignments = vec![SplitSlot::Test; record_count];
936        for &idx in indices.iter().take(n_tr) {
937            assignments[idx] = SplitSlot::Train;
938        }
939        for &idx in indices.iter().skip(n_tr).take(n_va) {
940            assignments[idx] = SplitSlot::Val;
941        }
942
943        if let Err(e) = std::fs::create_dir_all(outdir) {
944            let _ = ui::write_error(
945                err,
946                &format!("Failed to create directory {}: {}", outdir, e),
947            );
948            return Some(2);
949        }
950
951        if trace_stream {
952            let _ = ui::write_error(
953                err,
954                &format!("Streaming dataset input (records={})", record_count),
955            );
956        }
957
958        let out_root = std::path::Path::new(outdir);
959        let mut train_writer =
960            BufWriter::new(std::fs::File::create(out_root.join("train.jsonl")).unwrap());
961        let mut val_writer =
962            BufWriter::new(std::fs::File::create(out_root.join("val.jsonl")).unwrap());
963        let mut test_writer =
964            BufWriter::new(std::fs::File::create(out_root.join("test.jsonl")).unwrap());
965
966        let data_file = match std::fs::File::open(input) {
967            Ok(f) => f,
968            Err(e) => {
969                let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
970                return Some(2);
971            }
972        };
973        let reader = BufReader::new(data_file);
974        let mut record_idx = 0usize;
975        let mut first_line = true;
976
977        for (line_idx, line_res) in reader.lines().enumerate() {
978            let mut line = match line_res {
979                Ok(line) => line,
980                Err(e) => {
981                    let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
982                    return Some(2);
983                }
984            };
985            if first_line {
986                strip_utf8_bom(&mut line);
987                first_line = false;
988            }
989            if line.trim().is_empty() {
990                continue;
991            }
992            if let Err(e) = serde_json::from_str::<axm_engine::logger::HandRecord>(&line) {
993                let _ = ui::write_error(
994                    err,
995                    &format!("Invalid record at line {}: {}", line_idx + 1, e),
996                );
997                return Some(2);
998            }
999            let bucket = assignments
1000                .get(record_idx)
1001                .copied()
1002                .unwrap_or(SplitSlot::Test);
1003            record_idx += 1;
1004            match bucket {
1005                SplitSlot::Train => {
1006                    let _ = writeln!(train_writer, "{}", line);
1007                }
1008                SplitSlot::Val => {
1009                    let _ = writeln!(val_writer, "{}", line);
1010                }
1011                SplitSlot::Test => {
1012                    let _ = writeln!(test_writer, "{}", line);
1013                }
1014            }
1015        }
1016
1017        Some(0)
1018    }

1020    /// Aggregates statistics from JSONL hand history files.
1021    ///
1022    /// Reads hand history files (JSONL or .jsonl.zst) and computes summary statistics
1023    /// including total hands played and win distribution by player.
1024    ///
1025    /// # Arguments
1026    ///
1027    /// * `input` - Path to JSONL file or directory containing hand histories
1028    /// * `out` - Output stream for statistics report
1029    /// * `err` - Output stream for error messages and warnings
1030    ///
1031    /// # Returns
1032    ///
1033    /// Exit code: `0` on success, `2` if errors detected
1034    ///
1035    /// # Validation
1036    ///
1037    /// - Detects corrupted or incomplete records
1038    /// - Verifies chip conservation laws (sum of net_result must be zero)
1039    /// - Reports warnings for skipped records
1040    ///
1041    /// # Example
1042    ///
1043    /// ```no_run
1044    /// # use std::io;
1045    /// let input = "data/hands/sample.jsonl";
1046    /// let code = axm_cli::run(
1047    ///     vec!["axm", "stats", "--input", input],
1048    ///     &mut io::stdout(),
1049    ///     &mut io::stderr()
1050    /// );
1051    /// assert_eq!(code, 0);
1052    /// ```
1053    fn run_stats(input: &str, out: &mut dyn Write, err: &mut dyn Write) -> i32 {
1054        use std::path::Path;
1055
1056        struct StatsState {
1057            hands: u64,
1058            p0: u64,
1059            p1: u64,
1060            skipped: u64,
1061            corrupted: u64,
1062            stats_ok: bool,
1063        }
1064
1065        fn consume_stats_content(content: String, state: &mut StatsState, err: &mut dyn Write) {
1066            let has_trailing_nl = content.ends_with('\n');
1067            let lines: Vec<&str> = content.lines().filter(|l| !l.trim().is_empty()).collect();
1068            for (i, line) in lines.iter().enumerate() {
1069                let parsed: serde_json::Value = match serde_json::from_str(line) {
1070                    Ok(v) => v,
1071                    Err(_) => {
1072                        if i == lines.len() - 1 && !has_trailing_nl {
1073                            state.skipped += 1;
1074                        } else {
1075                            state.corrupted += 1;
1076                        }
1077                        continue;
1078                    }
1079                };
1080
1081                let rec: axm_engine::logger::HandRecord =
1082                    match serde_json::from_value(parsed.clone()) {
1083                        Ok(v) => v,
1084                        Err(_) => {
1085                            state.corrupted += 1;
1086                            continue;
1087                        }
1088                    };
1089
1090                if let Some(net_obj) = parsed.get("net_result").and_then(|v| v.as_object()) {
1091                    let mut sum = 0i64;
1092                    let mut invalid = false;
1093                    for (player, val) in net_obj {
1094                        if let Some(n) = val.as_i64() {
1095                            sum += n;
1096                        } else {
1097                            invalid = true;
1098                            state.stats_ok = false;
1099                            let _ = ui::write_error(
1100                                err,
1101                                &format!(
1102                                    "Invalid net_result value for {} at hand {}",
1103                                    player, rec.hand_id
1104                                ),
1105                            );
1106                        }
1107                    }
1108                    if sum != 0 {
1109                        state.stats_ok = false;
1110                        let _ = ui::write_error(
1111                            err,
1112                            &format!("Chip conservation violated at hand {}", rec.hand_id),
1113                        );
1114                    }
1115                    if invalid {
1116                        continue;
1117                    }
1118                }
1119
1120                state.hands += 1;
1121                if let Some(r) = rec.result.as_deref() {
1122                    if r == "p0" {
1123                        state.p0 += 1;
1124                    }
1125                    if r == "p1" {
1126                        state.p1 += 1;
1127                    }
1128                }
1129            }
1130        }
1131
1132        let path = Path::new(input);
1133        let mut state = StatsState {
1134            hands: 0,
1135            p0: 0,
1136            p1: 0,
1137            skipped: 0,
1138            corrupted: 0,
1139            stats_ok: true,
1140        };
1141
1142        if path.is_dir() {
1143            let mut stack = vec![path.to_path_buf()];
1144            while let Some(d) = stack.pop() {
1145                let rd = match std::fs::read_dir(&d) {
1146                    Ok(v) => v,
1147                    Err(_) => continue,
1148                };
1149                for entry in rd.flatten() {
1150                    let p = entry.path();
1151                    if p.is_dir() {
1152                        stack.push(p);
1153                        continue;
1154                    }
1155                    let name = p.file_name().and_then(|s| s.to_str()).unwrap_or("");
1156                    if !(name.ends_with(".jsonl") || name.ends_with(".jsonl.zst")) {
1157                        continue;
1158                    }
1159                    match read_text_auto(p.to_str().unwrap()) {
1160                        Ok(s) => consume_stats_content(s, &mut state, err),
1161                        Err(e) => {
1162                            let _ = ui::write_error(
1163                                err,
1164                                &format!("Failed to read {}: {}", p.display(), e),
1165                            );
1166                            state.stats_ok = false;
1167                        }
1168                    }
1169                }
1170            }
1171        } else {
1172            match read_text_auto(input) {
1173                Ok(s) => consume_stats_content(s, &mut state, err),
1174                Err(e) => {
1175                    let _ = ui::write_error(err, &format!("Failed to read {}: {}", input, e));
1176                    return 2;
1177                }
1178            }
1179        }
1180
1181        if state.corrupted > 0 {
1182            let _ = ui::write_error(
1183                err,
1184                &format!("Skipped {} corrupted record(s)", state.corrupted),
1185            );
1186        }
1187        if state.skipped > 0 {
1188            let _ = ui::write_error(
1189                err,
1190                &format!("Discarded {} incomplete final line(s)", state.skipped),
1191            );
1192        }
1193        if !path.is_dir() && state.hands == 0 && (state.corrupted > 0 || state.skipped > 0) {
1194            let _ = ui::write_error(err, "Invalid record");
1195            return 2;
1196        }
1197
1198        let summary = serde_json::json!({
1199            "hands": state.hands,
1200            "winners": { "p0": state.p0, "p1": state.p1 },
1201        });
1202        let _ = writeln!(out, "{}", serde_json::to_string_pretty(&summary).unwrap());
1203        if state.stats_ok {
1204            0
1205        } else {
1206            2
1207        }
1208    }

1210    fn export_sqlite(content: &str, output: &str, err: &mut dyn Write) -> i32 {
1211        enum ExportAttemptError {
1212            Busy(String),
1213            Fatal(String),
1214        }
1215
1216        fn sqlite_busy(err: &rusqlite::Error) -> bool {
1217            matches!(
1218                err,
1219                rusqlite::Error::SqliteFailure(info, _)
1220                    if matches!(
1221                        info.code,
1222                        rusqlite::ErrorCode::DatabaseBusy | rusqlite::ErrorCode::DatabaseLocked
1223                    )
1224            )
1225        }
1226
1227        fn export_sqlite_attempt(content: &str, output: &str) -> Result<(), ExportAttemptError> {
1228            let output_path = std::path::Path::new(output);
1229            if let Some(parent) = output_path.parent() {
1230                if !parent.as_os_str().is_empty() {
1231                    std::fs::create_dir_all(parent).map_err(|e| {
1232                        ExportAttemptError::Fatal(format!(
1233                            "Failed to create directory {}: {}",
1234                            parent.display(),
1235                            e
1236                        ))
1237                    })?;
1238                }
1239            }
1240
1241            let mut conn = rusqlite::Connection::open(output).map_err(|e| {
1242                if sqlite_busy(&e) {
1243                    ExportAttemptError::Busy(format!("open {}: {}", output, e))
1244                } else {
1245                    ExportAttemptError::Fatal(format!("Failed to open {}: {}", output, e))
1246                }
1247            })?;
1248
1249            let tx = conn.transaction().map_err(|e| {
1250                if sqlite_busy(&e) {
1251                    ExportAttemptError::Busy(format!("start transaction: {}", e))
1252                } else {
1253                    ExportAttemptError::Fatal(format!("Failed to start transaction: {}", e))
1254                }
1255            })?;
1256
1257            tx.execute("DROP TABLE IF EXISTS hands", []).map_err(|e| {
1258                if sqlite_busy(&e) {
1259                    ExportAttemptError::Busy(format!("reset schema: {}", e))
1260                } else {
1261                    ExportAttemptError::Fatal(format!("Failed to reset schema: {}", e))
1262                }
1263            })?;
1264
1265            tx.execute(
1266                "CREATE TABLE hands (
1267                hand_id TEXT PRIMARY KEY NOT NULL,
1268                seed INTEGER,
1269                result TEXT,
1270                ts TEXT,
1271                actions INTEGER NOT NULL,
1272                board INTEGER NOT NULL,
1273                raw_json TEXT NOT NULL
1274            )",
1275                [],
1276            )
1277            .map_err(|e| {
1278                if sqlite_busy(&e) {
1279                    ExportAttemptError::Busy(format!("create schema: {}", e))
1280                } else {
1281                    ExportAttemptError::Fatal(format!("Failed to create schema: {}", e))
1282                }
1283            })?;
1284
1285            let mut stmt = tx
1286                .prepare(
1287                    "INSERT INTO hands (hand_id, seed, result, ts, actions, board, raw_json)
1288             VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)",
1289                )
1290                .map_err(|e| {
1291                    if sqlite_busy(&e) {
1292                        ExportAttemptError::Busy(format!("prepare insert: {}", e))
1293                    } else {
1294                        ExportAttemptError::Fatal(format!("Failed to prepare insert: {}", e))
1295                    }
1296                })?;
1297
1298            for (line_idx, line) in content.lines().enumerate() {
1299                let raw = line.trim();
1300                if raw.is_empty() {
1301                    continue;
1302                }
1303
1304                let record: axm_engine::logger::HandRecord = serde_json::from_str(raw)
1305                    .map_err(|e| ExportAttemptError::Fatal(format!("Invalid record: {}", e)))?;
1306
1307                let axm_engine::logger::HandRecord {
1308                    hand_id,
1309                    seed,
1310                    actions,
1311                    board,
1312                    result,
1313                    ts,
1314                    ..
1315                } = record;
1316
1317                let actions_len = actions.len() as i64;
1318                let board_len = board.len() as i64;
1319
1320                let seed_val = match seed {
1321                    Some(v) if v > i64::MAX as u64 => {
1322                        return Err(ExportAttemptError::Fatal(format!(
1323                            "Seed {} exceeds supported range",
1324                            v
1325                        )));
1326                    }
1327                    Some(v) => Some(v as i64),
1328                    None => None,
1329                };
1330
1331                stmt.execute(rusqlite::params![
1332                    hand_id,
1333                    seed_val,
1334                    result,
1335                    ts,
1336                    actions_len,
1337                    board_len,
1338                    raw,
1339                ])
1340                .map_err(|e| {
1341                    if sqlite_busy(&e) {
1342                        ExportAttemptError::Busy(format!(
1343                            "insert record at line {}: {}",
1344                            line_idx + 1,
1345                            e
1346                        ))
1347                    } else {
1348                        ExportAttemptError::Fatal(format!("Failed to insert record: {}", e))
1349                    }
1350                })?;
1351            }
1352
1353            drop(stmt);
1354
1355            tx.commit().map_err(|e| {
1356                if sqlite_busy(&e) {
1357                    ExportAttemptError::Busy(format!("commit export: {}", e))
1358                } else {
1359                    ExportAttemptError::Fatal(format!("Failed to commit export: {}", e))
1360                }
1361            })?;
1362
1363            Ok(())
1364        }
1365
1366        let max_attempts = std::env::var("AXM_EXPORT_SQLITE_RETRIES")
1367            .ok()
1368            .and_then(|v| v.parse::<u32>().ok())
1369            .filter(|&v| v > 0)
1370            .unwrap_or(3);

2409    total: usize,
2410    level: u8,
2411    seed: Option<u64>,
2412    base_seed: u64,
2413    break_after: Option<usize>,
2414    per_hand_delay: Option<std::time::Duration>,
2415    mut completed: usize,
2416    path: Option<&std::path::Path>,
2417    out: &mut dyn Write,
2418    err: &mut dyn Write,

2421        Some(p) => match std::fs::OpenOptions::new()

2436        let mut engine = Engine::new(Some(base_seed + i as u64), level);
2437        engine.shuffle();
2438        let _ = engine.deal_hand();
2439
2440        if let Some(w) = writer.as_mut() {
2441            let hand_id = format!("19700101-{:06}", i + 1);
