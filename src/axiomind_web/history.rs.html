6/// Hand history storage and retrieval system
7#[derive(Debug)]
8pub struct HistoryStore {
9    hands: RwLock<Vec<HandRecord>>,

12impl HistoryStore {
13    pub fn new() -> Self {
14        Self {
15            hands: RwLock::new(Vec::new()),
16        }

19    /// Add a hand record to the history
20    pub fn add_hand(&self, record: HandRecord) -> Result<(), HistoryError> {
21        let mut hands = self
22            .hands
23            .write()
24            .map_err(|_| HistoryError::StoragePoisoned)?;
25        hands.push(record);
26        Ok(())
27    }
28
29    /// Get recent hands with optional limit
30    pub fn get_recent_hands(&self, limit: Option<usize>) -> Result<Vec<HandRecord>, HistoryError> {
31        let hands = self
32            .hands
33            .read()
34            .map_err(|_| HistoryError::StoragePoisoned)?;
35        let limit = limit.unwrap_or(100);
36        Ok(hands.iter().rev().take(limit).cloned().collect())
37    }
38
39    /// Get a specific hand by ID
40    pub fn get_hand(&self, hand_id: &str) -> Result<Option<HandRecord>, HistoryError> {
41        let hands = self
42            .hands
43            .read()
44            .map_err(|_| HistoryError::StoragePoisoned)?;
45        Ok(hands.iter().find(|h| h.hand_id == hand_id).cloned())
46    }
47
48    /// Filter hands by criteria
49    pub fn filter_hands(&self, filter: HandFilter) -> Result<Vec<HandRecord>, HistoryError> {
50        let hands = self
51            .hands
52            .read()
53            .map_err(|_| HistoryError::StoragePoisoned)?;
54        let filtered: Vec<HandRecord> = hands
55            .iter()
56            .filter(|h| filter.matches(h))
57            .cloned()
58            .collect();
59        Ok(filtered)
60    }

62    /// Calculate statistics from stored hands
63    pub fn calculate_stats(&self) -> Result<HandStatistics, HistoryError> {
64        let hands = self
65            .hands
66            .read()
67            .map_err(|_| HistoryError::StoragePoisoned)?;
68
69        if hands.is_empty() {
70            return Ok(HandStatistics::default());
71        }
72
73        let total_hands = hands.len();
74        let mut wins = 0;
75        let mut total_pot = 0;
76
77        for hand in hands.iter() {
78            // Count wins (simplified - check if result indicates player 0 wins)
79            if let Some(result) = &hand.result
80                && result.contains("player 0 wins")
81            {
82                wins += 1;
83            }
84
85            // Estimate pot size from actions (simplified)
86            total_pot += estimate_pot_size(hand);
87        }
88
89        let win_rate = (wins as f64 / total_hands as f64) * 100.0;
90        let avg_pot = total_pot as f64 / total_hands as f64;
91
92        Ok(HandStatistics {
93            total_hands,
94            wins,
95            win_rate,
96            avg_pot_size: avg_pot,
97        })
98    }

100    /// Get the total number of hands
101    pub fn total_hands(&self) -> Result<usize, HistoryError> {

116/// Filter criteria for hand history
117#[derive(Debug, Clone, Default, Serialize, Deserialize)]
118pub struct HandFilter {
119    #[serde(skip_serializing_if = "Option::is_none")]
120    pub result_type: Option<String>, // "win", "loss", "tie"
121    #[serde(skip_serializing_if = "Option::is_none")]
122    pub date_from: Option<String>,
123    #[serde(skip_serializing_if = "Option::is_none")]
124    pub date_to: Option<String>,
125}
126
127impl HandFilter {
128    fn matches(&self, hand: &HandRecord) -> bool {
129        // Filter by result type
130        if let Some(result_type) = &self.result_type {
131            if let Some(result) = &hand.result {
132                if !result.to_lowercase().contains(&result_type.to_lowercase()) {
133                    return false;
134                }
135            } else {
136                return false;
137            }
138        }
139
140        // Filter by date range
141        if let (Some(ts), Some(date_from)) = (&hand.ts, &self.date_from)
142            && ts < date_from
143        {
144            return false;
145        }
146
147        if let (Some(ts), Some(date_to)) = (&hand.ts, &self.date_to)
148            && ts > date_to
149        {
150            return false;
151        }
152
153        true
154    }
155}

184#[derive(Debug, Error)]
185pub enum HistoryError {
186    #[error("History storage poisoned")]
187    StoragePoisoned,
188    #[error("Hand not found: {0}")]
189    NotFound(String),
190}
191
192impl crate::errors::IntoErrorResponse for HistoryError {
193    fn status_code(&self) -> warp::http::StatusCode {
194        use warp::http::StatusCode;
195        match self {
196            HistoryError::StoragePoisoned => StatusCode::INTERNAL_SERVER_ERROR,
197            HistoryError::NotFound(_) => StatusCode::NOT_FOUND,
198        }
199    }
200
201    fn error_code(&self) -> &'static str {
202        match self {
203            HistoryError::StoragePoisoned => "history_storage_error",
204            HistoryError::NotFound(_) => "hand_not_found",
205        }
206    }
207
208    fn error_message(&self) -> String {
209        self.to_string()
210    }
211
212    fn severity(&self) -> crate::errors::ErrorSeverity {
213        use crate::errors::ErrorSeverity;
214        match self {
215            HistoryError::StoragePoisoned => ErrorSeverity::Critical,
216            HistoryError::NotFound(_) => ErrorSeverity::Client,
217        }
218    }
219}
