3/// Represents one of the four suits in a standard 52-card deck.
4/// Used as a component of [`Card`] to fully define a playing card.
5#[derive(Debug, Copy, Clone, Eq, PartialEq, Hash, Ord, PartialOrd, Serialize, Deserialize)]
6pub enum Suit {
7    /// Clubs suit (♣)
8    Clubs,
9    /// Diamonds suit (♦)
10    Diamonds,
11    /// Hearts suit (♥)
12    Hearts,
13    /// Spades suit (♠)
14    Spades,
15}
16
17/// Represents the rank (face value) of a playing card from Two through Ace.
18/// Numeric values are assigned for comparison and hand evaluation purposes.
19#[derive(Debug, Copy, Clone, Eq, PartialEq, Hash, Ord, PartialOrd, Serialize, Deserialize)]
20pub enum Rank {
21    /// Rank 2
22    Two = 2,
23    /// Rank 3
24    Three,
25    /// Rank 4
26    Four,
27    /// Rank 5
28    Five,
29    /// Rank 6
30    Six,
31    /// Rank 7
32    Seven,
33    /// Rank 8
34    Eight,
35    /// Rank 9
36    Nine,
37    /// Rank 10
38    Ten,
39    /// Jack (11)
40    Jack,
41    /// Queen (12)
42    Queen,
43    /// King (13)
44    King,
45    /// Ace (14)
46    Ace,
47}
48
49impl Rank {
50    pub fn from_u8(v: u8) -> Rank {
51        match v {
52            2 => Rank::Two,
53            3 => Rank::Three,
54            4 => Rank::Four,
55            5 => Rank::Five,
56            6 => Rank::Six,
57            7 => Rank::Seven,
58            8 => Rank::Eight,
59            9 => Rank::Nine,
60            10 => Rank::Ten,
61            11 => Rank::Jack,
62            12 => Rank::Queen,
63            13 => Rank::King,
64            _ => Rank::Ace,
65        }
66    }
67}

69/// Represents a single playing card with a suit and rank.
70/// Cards are the fundamental unit of the poker game, used in player hands, the board, and the deck.
71#[derive(Debug, Copy, Clone, Eq, PartialEq, Hash, Ord, PartialOrd, Serialize, Deserialize)]
72pub struct Card {
73    /// The suit of the card (Clubs, Diamonds, Hearts, or Spades)
74    pub suit: Suit,
75    /// The rank of the card (Two through Ace)
76    pub rank: Rank,
77}
78
79pub fn all_suits() -> [Suit; 4] {
80    [Suit::Clubs, Suit::Diamonds, Suit::Hearts, Suit::Spades]
81}
82
83pub fn all_ranks() -> [Rank; 13] {
84    [
85        Rank::Two,
86        Rank::Three,
87        Rank::Four,
88        Rank::Five,
89        Rank::Six,
90        Rank::Seven,
91        Rank::Eight,
92        Rank::Nine,
93        Rank::Ten,
94        Rank::Jack,
95        Rank::Queen,
96        Rank::King,
97        Rank::Ace,
98    ]
99}
100
101pub fn full_deck() -> Vec<Card> {
102    let mut v = Vec::with_capacity(52);
103    for &s in &all_suits() {
104        for &r in &all_ranks() {
105            v.push(Card { suit: s, rank: r });
106        }
107    }
108    v
