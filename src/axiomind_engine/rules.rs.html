4#[derive(Debug, Clone, PartialEq, Eq)]
5pub enum ValidatedAction {
6    Fold,
7    Check,
8    Call(u32),
9    Bet(u32),
10    Raise(u32),
11    AllIn(u32),
12}
13
14/// Validates a player action according to betting rules and stack size.
15///
16/// Converts a [`crate::player::PlayerAction`] into a [`ValidatedAction`], enforcing poker
17/// betting rules, minimum raise amounts, and all-in logic when the player
18/// doesn't have enough chips.
19///
20/// # Arguments
21///
22/// * `stack` - Player's remaining chip stack
23/// * `to_call` - Amount needed to call the current bet
24/// * `min_raise` - Minimum allowed raise amount (typically previous raise size)
25/// * `action` - The action the player wishes to perform
26///
27/// # Returns
28///
29/// Returns `Ok(ValidatedAction)` if the action is legal, containing the
30/// actual action to execute (which may be converted to AllIn if stack is insufficient).
31///
32/// # Errors
33///
34/// Returns [`GameError`] in the following cases:
35/// - [`GameError::InsufficientChips`] - Player tries to check when facing a bet
36/// - [`GameError::InvalidBetAmount`] - Bet/raise amount is below minimum or zero
37///
38/// # Examples
39///
40/// ```
41/// use axiomind_engine::rules::{validate_action, ValidatedAction};
42/// use axiomind_engine::player::PlayerAction;
43///
44/// // Valid call with sufficient stack
45/// let result = validate_action(1000, 50, 100, PlayerAction::Call);
46/// assert!(matches!(result, Ok(ValidatedAction::Call(50))));
47///
48/// // All-in when stack is insufficient for full raise
49/// let result = validate_action(80, 50, 100, PlayerAction::Raise(100));
50/// assert!(matches!(result, Ok(ValidatedAction::AllIn(80))));
51/// ```
52///
53/// ```
54/// use axiomind_engine::rules::validate_action;
55/// use axiomind_engine::player::PlayerAction;
56/// use axiomind_engine::errors::GameError;
57///
58/// // Invalid: check when facing a bet
59/// let result = validate_action(1000, 50, 100, PlayerAction::Check);
60/// assert!(matches!(result, Err(GameError::InsufficientChips)));
61///
62/// // Invalid: raise below minimum
63/// let result = validate_action(1000, 50, 100, PlayerAction::Raise(50));
64/// assert!(matches!(result, Err(GameError::InvalidBetAmount { .. })));
65/// ```
66pub fn validate_action(
67    stack: u32,
68    to_call: u32,
69    min_raise: u32,
70    action: A,
71) -> Result<ValidatedAction, GameError> {
72    match action {
73        A::Fold => Ok(ValidatedAction::Fold),
74        A::Check => {
75            if to_call == 0 {
76                Ok(ValidatedAction::Check)
77            } else {
78                Err(GameError::InsufficientChips)
79            }
80        }
81        A::Call => {
82            if stack <= to_call {
83                Ok(ValidatedAction::AllIn(stack))
84            } else {
85                Ok(ValidatedAction::Call(to_call))
86            }
87        }
88        A::Bet(amount) => {
89            if amount == 0 {
90                return Err(GameError::InvalidBetAmount { amount, minimum: 1 });
91            }
92            if amount >= stack {
93                Ok(ValidatedAction::AllIn(stack))
94            } else {
95                Ok(ValidatedAction::Bet(amount))
96            }
97        }
98        A::Raise(amount) => {
99            if amount + to_call >= stack {
100                Ok(ValidatedAction::AllIn(stack))
101            } else if amount < min_raise {
102                Err(GameError::InvalidBetAmount {
103                    amount,
104                    minimum: min_raise,
105                })
106            } else {
107                Ok(ValidatedAction::Raise(amount))
108            }
109        }
110        A::AllIn => Ok(ValidatedAction::AllIn(stack)),
111    }
112}
